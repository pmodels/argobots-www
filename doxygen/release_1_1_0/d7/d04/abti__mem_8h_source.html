255 ABTU_ret_err static inline int ABTI_mem_alloc_ythread_mempool_desc(
256  ABTI_local *p_local, ABTI_thread_attr *p_attr, ABTI_ythread **pp_ythread)
257 {
258  ABTI_ythread *p_ythread;
259  if (sizeof(ABTI_ythread) <= ABTI_MEM_POOL_DESC_ELEM_SIZE) {
260  /* Use a descriptor pool for ABT_thread. */
261  ABTI_STATIC_ASSERT(offsetof(ABTI_ythread, thread) == 0);
262  int abt_errno =
263  ABTI_mem_alloc_nythread(p_local, (ABTI_thread **)&p_ythread);
264  ABTI_CHECK_ERROR(abt_errno);
265  } else {
266  /* Do not allocate stack, but Valgrind registration is preferred. */
267  int abt_errno = ABTU_malloc(sizeof(ABTI_ythread), (void **)&p_ythread);
268  ABTI_CHECK_ERROR(abt_errno);
269  p_ythread->thread.type = ABTI_THREAD_TYPE_MEM_MALLOC_DESC;
270  }
271  /* Copy members of p_attr. */
272  p_ythread->stacksize = p_attr->stacksize;
273  p_ythread->p_stack = p_attr->p_stack;
274  /* Note that the valgrind registration is ignored iff p_stack is NULL. */
275  ABTI_mem_register_stack(p_ythread->p_stack, p_ythread->stacksize);
276  *pp_ythread = p_ythread;
277  return ABT_SUCCESS;

329 ABTU_ret_err static inline int ABTI_mem_alloc_desc(ABTI_local *p_local,
330  void **pp_desc)
331 {
332 #ifndef ABT_CONFIG_USE_MEM_POOL
333  return ABTU_malloc(ABTI_MEM_POOL_DESC_SIZE, pp_desc);
334 #else
335  void *p_desc;
336  ABTI_xstream *p_local_xstream = ABTI_local_get_xstream_or_null(p_local);
337  if (ABTI_IS_EXT_THREAD_ENABLED && p_local_xstream == NULL) {
338  /* For external threads */
339  int abt_errno = ABTU_malloc(ABTI_MEM_POOL_DESC_SIZE, &p_desc);
340  ABTI_CHECK_ERROR(abt_errno);
341  *(uint32_t *)(((char *)p_desc) + ABTI_MEM_POOL_DESC_SIZE) = 1;
342  *pp_desc = p_desc;

344  } else {

348  ABTI_CHECK_ERROR(abt_errno);
