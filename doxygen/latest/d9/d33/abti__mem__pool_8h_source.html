113 static inline void *ABTI_mem_pool_alloc(ABTI_mem_pool_local_pool *p_local_pool)
114 {
115  size_t bucket_index = p_local_pool->bucket_index;
116  ABTI_mem_pool_header *cur_bucket = p_local_pool->buckets[bucket_index];
117  int num_headers_in_cur_bucket = cur_bucket->bucket_info.num_headers;
118  /* At least one header is available in the current bucket, so it must be
119  * larger than 0. */
120  ABTI_ASSERT(num_headers_in_cur_bucket >= 1);
121  if (num_headers_in_cur_bucket == 1) {
122  /*cur_bucket will be empty after allocation. */
123  if (bucket_index == 0) {
124  /* cur_bucket is the last header in this pool.
125  * Let's get some buckets from the global pool. */
126  int i;
127  for (i = 0; i < ABT_MEM_POOL_NUM_TAKE_BUCKETS; i++) {
128  p_local_pool->buckets[i] =
129  ABTI_mem_pool_take_bucket(p_local_pool->p_global_pool);
130  }
131  p_local_pool->bucket_index = ABT_MEM_POOL_NUM_TAKE_BUCKETS - 1;
132  } else {
133  p_local_pool->bucket_index = bucket_index - 1;
134  }
135  /* Now buckets[bucket_index] is full of headers. */
136  } else {
137  /* Let's return the header in the bucket. */
138  ABTI_mem_pool_header *p_next = cur_bucket->p_next;
139  p_next->bucket_info.num_headers = num_headers_in_cur_bucket - 1;
140  p_local_pool->buckets[bucket_index] = p_next;
141  }
142  /* At least one header is available in the current bucket. */
143  return (void *)cur_bucket;
144 }

146 static inline void ABTI_mem_pool_free(ABTI_mem_pool_local_pool *p_local_pool,
147  void *mem)
148 {
149  /* At least one header is available in the current bucket. */
150  size_t bucket_index = p_local_pool->bucket_index;
151  ABTI_mem_pool_header *p_freed_header = (ABTI_mem_pool_header *)mem;
152  ABTI_mem_pool_header *cur_bucket = p_local_pool->buckets[bucket_index];
153  if (cur_bucket->bucket_info.num_headers ==
154  p_local_pool->num_headers_per_bucket) {
155  /* cur_bucket is full. */
156  if (++bucket_index == ABT_MEM_POOL_MAX_LOCAL_BUCKETS) {
157  int i;
158  /* All buckets are full, so let's return some old buckets. */
159  for (i = 0; i < ABT_MEM_POOL_NUM_RETURN_BUCKETS; i++) {
160  ABTI_mem_pool_return_bucket(p_local_pool->p_global_pool,
161  p_local_pool->buckets[i]);
162  }
163  for (i = ABT_MEM_POOL_NUM_RETURN_BUCKETS;
164  i < ABT_MEM_POOL_MAX_LOCAL_BUCKETS; i++) {
165  p_local_pool->buckets[i - ABT_MEM_POOL_NUM_RETURN_BUCKETS] =
166  p_local_pool->buckets[i];
167  }
168  bucket_index = ABT_MEM_POOL_MAX_LOCAL_BUCKETS -
169  ABT_MEM_POOL_NUM_RETURN_BUCKETS;
170  }
171  p_local_pool->bucket_index = bucket_index;
172  p_freed_header->p_next = NULL;
173  p_freed_header->bucket_info.num_headers = 1;
174  } else {
175  p_freed_header->p_next = cur_bucket;
176  p_freed_header->bucket_info.num_headers =
177  cur_bucket->bucket_info.num_headers + 1;
178  }
179  p_local_pool->buckets[bucket_index] = p_freed_header;
180  /* At least one header is available in the current bucket. */
181 }
