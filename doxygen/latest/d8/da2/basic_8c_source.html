90 {

92  ABT_unit unit = ABT_UNIT_NULL;

95  uint32_t event_freq;

98  int i;
99 
100  ABTI_sched *p_sched = ABTI_sched_get_ptr(sched);
101  ABTI_ASSERT(p_sched);
102 
103  p_data = sched_data_get_ptr(p_sched->data);
104  event_freq = p_data->event_freq;
105  num_pools = p_data->num_pools;
106  pools = p_data->pools;
107 
108  while (1) {
109  for (i = 0; i < num_pools; i++) {
110  ABTI_pool *p_pool = ABTI_pool_get_ptr(pools[i]);
111  ++pop_count;
112  if ((unit = ABTI_pool_pop(p_pool)) != ABT_UNIT_NULL) {
113  ABTI_xstream_run_unit(&p_local_xstream, unit, p_pool);
114  break;
115  }
116  }
117  /* if we attempted event_freq pops, check for events */
118  if (pop_count >= event_freq) {
119  ABTI_xstream_check_events(p_local_xstream, sched);
120  if (ABTI_sched_has_to_stop(&p_local_xstream, p_sched) == ABT_TRUE)
121  break;
122  SCHED_SLEEP(unit != ABT_UNIT_NULL, p_data->sleep_time);
123  pop_count = 0;
124  }
125  }
126 }
