1806 static inline void xstream_schedule_ythread(ABTI_global *p_global,
1807  ABTI_xstream **pp_local_xstream,
1808  ABTI_ythread *p_ythread)
1809 {
1810  ABTI_xstream *p_local_xstream = *pp_local_xstream;
1811 
1812 #ifndef ABT_CONFIG_DISABLE_THREAD_CANCEL
1813  if (ABTD_atomic_acquire_load_uint32(&p_ythread->thread.request) &
1814  ABTI_THREAD_REQ_CANCEL) {
1815  LOG_DEBUG("[U%" PRIu64 ":E%d] canceled\n",
1816  ABTI_thread_get_id(&p_ythread->thread),
1817  p_local_xstream->rank);
1818  ABTD_ythread_cancel(p_local_xstream, p_ythread);
1819  ABTI_xstream_terminate_thread(p_global,
1820  ABTI_xstream_get_local(p_local_xstream),
1821  &p_ythread->thread);
1822  return;
1823  }
1824 #endif
1825 
1826 #ifndef ABT_CONFIG_DISABLE_MIGRATION
1827  if (ABTD_atomic_acquire_load_uint32(&p_ythread->thread.request) &
1828  ABTI_THREAD_REQ_MIGRATE) {
1829  int abt_errno =
1830  xstream_migrate_thread(p_global,
1831  ABTI_xstream_get_local(p_local_xstream),
1832  &p_ythread->thread);
1833  if (!ABTI_IS_ERROR_CHECK_ENABLED || abt_errno == ABT_SUCCESS) {
1834  /* Migration succeeded, so we do not need to schedule p_ythread. */
1835  return;
1836  }
1837  }
1838 #endif
1839 
1840  /* Change the last ES */
1841  p_ythread->thread.p_last_xstream = p_local_xstream;
1842 
1843  /* Change the ULT state */
1844  ABTD_atomic_release_store_int(&p_ythread->thread.state,
1845  ABT_THREAD_STATE_RUNNING);
1846 
1847  /* Switch the context */
1848  LOG_DEBUG("[U%" PRIu64 ":E%d] start running\n",
1849  ABTI_thread_get_id(&p_ythread->thread), p_local_xstream->rank);
1850 
1851  /* Since the argument is pp_local_xstream, p_local_xstream->p_thread must be
1852  * yieldable. */
1853  ABTI_ythread *p_self = ABTI_thread_get_ythread(p_local_xstream->p_thread);
1854  p_ythread = ABTI_ythread_context_switch_to_child(pp_local_xstream, p_self,
1855  p_ythread);
1856  /* The previous ULT (p_ythread) may not be the same as one to which the
1857  * context has been switched. */
1858  /* The scheduler continues from here. */
1859  p_local_xstream = *pp_local_xstream;
1860 
1861  LOG_DEBUG("[U%" PRIu64 ":E%d] stopped\n",
1862  ABTI_thread_get_id(&p_ythread->thread), p_local_xstream->rank);
1863 
1864  /* Request handling. */
1865  /* We do not need to acquire-load request since all critical requests
1866  * (BLOCK, ORPHAN, STOP, and NOPUSH) are written by p_ythread. CANCEL might
1867  * be delayed. */
1868  uint32_t request =
1869  ABTD_atomic_acquire_load_uint32(&p_ythread->thread.request);
1870  if (request & ABTI_THREAD_REQ_TERMINATE) {
1871  /* The ULT has completed its execution or it called the exit request. */
1872  LOG_DEBUG("[U%" PRIu64 ":E%d] finished\n",
1873  ABTI_thread_get_id(&p_ythread->thread),
1874  p_local_xstream->rank);
1875  ABTI_xstream_terminate_thread(p_global,
1876  ABTI_xstream_get_local(p_local_xstream),
1877  &p_ythread->thread);
1878 #ifndef ABT_CONFIG_DISABLE_THREAD_CANCEL
1879  } else if (request & ABTI_THREAD_REQ_CANCEL) {
1880  LOG_DEBUG("[U%" PRIu64 ":E%d] canceled\n",
1881  ABTI_thread_get_id(&p_ythread->thread),
1882  p_local_xstream->rank);
1883  ABTD_ythread_cancel(p_local_xstream, p_ythread);
1884  ABTI_xstream_terminate_thread(p_global,
1885  ABTI_xstream_get_local(p_local_xstream),
1886  &p_ythread->thread);
1887 #endif
1888  } else if (!(request & ABTI_THREAD_REQ_NON_YIELD)) {
1889  /* The ULT did not finish its execution.
1890  * Change the state of current running ULT and
1891  * add it to the pool again. */
1892  ABTI_pool_add_thread(&p_ythread->thread);
1893  } else if (request & ABTI_THREAD_REQ_BLOCK) {
1894  LOG_DEBUG("[U%" PRIu64 ":E%d] check blocked\n",
1895  ABTI_thread_get_id(&p_ythread->thread),
1896  p_local_xstream->rank);
1897  ABTI_thread_unset_request(&p_ythread->thread, ABTI_THREAD_REQ_BLOCK);
1898 #ifndef ABT_CONFIG_DISABLE_MIGRATION
1899  } else if (request & ABTI_THREAD_REQ_MIGRATE) {
1900  /* This is the case when the ULT requests migration of itself. */
1901  int abt_errno =
1902  xstream_migrate_thread(p_global,
1903  ABTI_xstream_get_local(p_local_xstream),
1904  &p_ythread->thread);
1905  /* Migration is optional, so it is okay if it fails. */
1906  (void)abt_errno;
1907 #endif
1908  } else if (request & ABTI_THREAD_REQ_ORPHAN) {
1909  /* The ULT is not pushed back to the pool and is disconnected from any
1910  * pool. */
1911  LOG_DEBUG("[U%" PRIu64 ":E%d] orphaned\n",
1912  ABTI_thread_get_id(&p_ythread->thread),
1913  p_local_xstream->rank);
1914  ABTI_thread_unset_request(&p_ythread->thread, ABTI_THREAD_REQ_ORPHAN);
1915  p_ythread->thread.p_pool->u_free(&p_ythread->thread.unit);
1916  p_ythread->thread.p_pool = NULL;
1917  } else if (request & ABTI_THREAD_REQ_NOPUSH) {
1918  /* The ULT is not pushed back to the pool */
1919  LOG_DEBUG("[U%" PRIu64 ":E%d] not pushed\n",
1920  ABTI_thread_get_id(&p_ythread->thread),
1921  p_local_xstream->rank);
1922  ABTI_thread_unset_request(&p_ythread->thread, ABTI_THREAD_REQ_NOPUSH);
1923  } else {
1924  ABTI_ASSERT(0);
1925  ABTU_unreachable();
1926  }
1927 }

1929 static inline void xstream_schedule_task(ABTI_global *p_global,
1930  ABTI_xstream *p_local_xstream,
1931  ABTI_thread *p_task)
1932 {
1933 #ifndef ABT_CONFIG_DISABLE_TASK_CANCEL
1934  if (ABTD_atomic_acquire_load_uint32(&p_task->request) &
1935  ABTI_THREAD_REQ_CANCEL) {
1936  ABTI_tool_event_thread_cancel(p_local_xstream, p_task);
1937  ABTI_xstream_terminate_thread(p_global,
1938  ABTI_xstream_get_local(p_local_xstream),
1939  p_task);
1940  return;
1941  }
1942 #endif
1943 
1944  /* Change the task state */
1945  ABTD_atomic_release_store_int(&p_task->state, ABT_THREAD_STATE_RUNNING);
1946 
1947  /* Set the associated ES */
1948  p_task->p_last_xstream = p_local_xstream;
1949
