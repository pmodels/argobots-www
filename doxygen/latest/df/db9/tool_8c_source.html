260  *(int *)val = depth - 1;
261  }
262  break;

264  if (!p_tctx->p_caller) {
265  *(ABT_exec_entity_type *)val = ABT_EXEC_ENTITY_TYPE_EXT;

268  }
269  break;

271  if (!p_tctx->p_caller) {
272  *(void **)val = NULL;

274  *(ABT_thread *)val = ABTI_thread_get_handle(p_tctx->p_caller);
