745

748  p_cur_ythread->thread.p_last_xstream->rank,
749  ABTI_thread_get_id(&p_tar_ythread->thread));
750

752  ABTI_CHECK_TRUE_MSG(p_cur_ythread != p_tar_ythread, ABT_ERR_INV_THREAD,

758  ABT_ERR_INV_THREAD,

1914  p_mig_data->p_migration_cb_arg = p_attr->p_cb_arg;

1925  }
1926  }
1927 #endif
1928  }

1939  void *p_stack = p_newthread->p_stack;

1951  ABTD_ythread_context_init(NULL, &p_newthread->ctx);

1956

1972  if (p_keytable)
1973  ABTI_ktable_free(p_local, p_keytable);
1974  ABTI_mem_free_thread(p_local, &p_newthread->thread);
1975  return abt_errno;
1976  }
1977  }
1978  ABTD_atomic_relaxed_store_ptr(&p_newthread->thread.p_keytable, p_keytable);

1982  if (thread_type & ABTI_THREAD_TYPE_MAIN) {
1983  LOG_DEBUG("[U%" PRIu64 "] main ULT created\n", thread_id);
1984  } else if (thread_type & ABTI_THREAD_TYPE_MAIN_SCHED) {
1985  LOG_DEBUG("[U%" PRIu64 "] main sched ULT created\n", thread_id);
1986  } else {
1987  LOG_DEBUG("[U%" PRIu64 "] created\n", thread_id);
1988  }
1989 #endif
1990

1992  ABTI_tool_event_thread_create(p_local, &p_newthread->thread,
1993  ABTI_local_get_xstream_or_null(p_local)
1994  ? ABTI_local_get_xstream(p_local)

1996  : NULL,
1997  push_pool ? p_pool : NULL);

2004  ABTI_pool_push(p_pool, p_newthread->thread.unit);

2006  p_newthread->thread.unit = ABT_UNIT_NULL;

2010  *pp_newthread = p_newthread;
2011  return ABT_SUCCESS;

2102 static inline void thread_join(ABTI_local **pp_local, ABTI_thread *p_thread)
2103 {
2104  if (ABTD_atomic_acquire_load_int(&p_thread->state) ==
2105  ABT_THREAD_STATE_TERMINATED) {
2106  goto fn_exit;
2107  }
2108  /* The main ULT cannot be joined. */
2109  ABTI_ASSERT(!(p_thread->type & ABTI_THREAD_TYPE_MAIN));
2110 
2111  ABTI_xstream *p_local_xstream = ABTI_local_get_xstream_or_null(*pp_local);
2112  if (ABTI_IS_EXT_THREAD_ENABLED && !p_local_xstream)
2113  goto busywait_based;
2114 
2115  ABTI_thread *p_self_thread = p_local_xstream->p_thread;
2116 
2117  ABTI_ythread *p_self = ABTI_thread_get_ythread_or_null(p_self_thread);
2118  if (!p_self)
2119  goto busywait_based;
2120 
2121  /* The target ULT should be different. */
2122  ABTI_ASSERT(p_thread != p_self_thread);
2123 
2124  ABTI_ythread *p_ythread = ABTI_thread_get_ythread_or_null(p_thread);
2125  if (!p_ythread)
2126  goto yield_based_task;
2127 
2128  ABT_pool_access access = p_self->thread.p_pool->access;
2129 
2130  if ((p_self->thread.p_pool == p_ythread->thread.p_pool) &&
2131  (access == ABT_POOL_ACCESS_PRIV || access == ABT_POOL_ACCESS_MPSC ||
2132  access == ABT_POOL_ACCESS_SPSC) &&
2133  (ABTD_atomic_acquire_load_int(&p_ythread->thread.state) ==
2134  ABT_THREAD_STATE_READY)) {
2135 
2136  ABTI_xstream *p_xstream = p_self->thread.p_last_xstream;
2137 
2138  /* If other ES is calling ABTI_ythread_set_ready(), p_ythread may not
2139  * have been added to the pool yet because ABTI_ythread_set_ready()
2140  * changes the state first followed by pushing p_ythread to the pool.
2141  * Therefore, we have to check whether p_ythread is in the pool, and if
2142  * not, we need to wait until it is added. */
2143  while (p_ythread->thread.p_pool->u_is_in_pool(p_ythread->thread.unit) !=
2144  ABT_TRUE) {
2145  }
2146 
2147  /* This is corresponding to suspension. */
2148  ABTI_tool_event_ythread_suspend(p_local_xstream, p_self,
2149  p_self->thread.p_parent,
2150  ABT_SYNC_EVENT_TYPE_THREAD_JOIN,
2151  (void *)p_ythread);
2152 
2153  /* Increase the number of blocked units. Be sure to execute
2154  * ABTI_pool_inc_num_blocked before ABTI_POOL_REMOVE in order not to
2155  * underestimate the number of units in a pool. */
2156  ABTI_pool_inc_num_blocked(p_self->thread.p_pool);
2157  /* Remove the target ULT from the pool */
2158  int abt_errno =
2159  ABTI_pool_remove(p_ythread->thread.p_pool, p_ythread->thread.unit);
2160  /* This failure is fatal. */
2161  ABTI_ASSERT(abt_errno == ABT_SUCCESS);
2162 
2163  /* Set the link in the context for the target ULT. Since p_link will be
2164  * referenced by p_self, this update does not require release store. */
2165  ABTD_atomic_relaxed_store_ythread_context_ptr(&p_ythread->ctx.p_link,
2166  &p_self->ctx);
2167  /* Set the last ES */
2168  p_ythread->thread.p_last_xstream = p_xstream;
2169  ABTD_atomic_release_store_int(&p_ythread->thread.state,
2170  ABT_THREAD_STATE_RUNNING);
2171 
2172  /* Make the current ULT BLOCKED */
2173  ABTD_atomic_release_store_int(&p_self->thread.state,
2174  ABT_THREAD_STATE_BLOCKED);
2175 
2176  LOG_DEBUG("[U%" PRIu64 ":E%d] blocked to join U%" PRIu64 "\n",
2177  ABTI_thread_get_id(&p_self->thread),
2178  p_self->thread.p_last_xstream->rank,
2179  ABTI_thread_get_id(&p_ythread->thread));
2180  LOG_DEBUG("[U%" PRIu64 ":E%d] start running\n",
2181  ABTI_thread_get_id(&p_ythread->thread),
2182  p_ythread->thread.p_last_xstream->rank);
2183 
2184  /* Switch the context */
2185  ABTI_ythread *p_prev =
