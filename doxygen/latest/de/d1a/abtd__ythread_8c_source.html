35 static inline void ABTD_ythread_terminate(ABTI_xstream *p_local_xstream,
36  ABTI_ythread *p_ythread)
37 {
38  ABTD_ythread_context *p_ctx = &p_ythread->ctx;
39  ABTD_ythread_context *p_link =
40  ABTD_atomic_acquire_load_ythread_context_ptr(&p_ctx->p_link);
41  if (p_link) {
42  /* If p_link is set, it means that other ULT has called the join. */
43  ABTI_ythread *p_joiner = ABTI_ythread_context_get_ythread(p_link);
44  if (p_ythread->thread.p_last_xstream ==
45  p_joiner->thread.p_last_xstream) {
46  /* Only when the current ULT is on the same ES as p_joiner's,
47  * we can jump to the joiner ULT. */
48  ABTD_atomic_release_store_int(&p_ythread->thread.state,
49  ABTI_THREAD_STATE_TERMINATED);
50  LOG_DEBUG("[U%" PRIu64 ":E%d] terminated\n",
51  ABTI_thread_get_id(&p_ythread->thread),
52  p_ythread->thread.p_last_xstream->rank);
53 
54  /* Note that a parent ULT cannot be a joiner. */
55  ABTI_tool_event_ythread_resume(ABTI_xstream_get_local(
56  p_local_xstream),
57  p_joiner, &p_ythread->thread);
58  ABTI_ythread_finish_context_to_sibling(p_local_xstream, p_ythread,
59  p_joiner);
60  return;
61  } else {
62  /* If the current ULT's associated ES is different from p_joiner's,
63  * we can't directly jump to p_joiner. Instead, we wake up
64  * p_joiner here so that p_joiner's scheduler can resume it. */
65  ABTI_ythread_set_ready(ABTI_xstream_get_local(p_local_xstream),
66  p_joiner);
67 
68  /* We don't need to use the atomic OR operation here because the ULT
69  * will be terminated regardless of other requests. */
70  ABTD_atomic_release_store_uint32(&p_ythread->thread.request,
71  ABTI_THREAD_REQ_TERMINATE);
72  }
73  } else {
74  uint32_t req =
75  ABTD_atomic_fetch_or_uint32(&p_ythread->thread.request,
76  ABTI_THREAD_REQ_JOIN |
77  ABTI_THREAD_REQ_TERMINATE);
78  if (req & ABTI_THREAD_REQ_JOIN) {
79  /* This case means there has been a join request and the joiner has
80  * blocked. We have to wake up the joiner ULT. */
81  do {
82  p_link = ABTD_atomic_acquire_load_ythread_context_ptr(
83  &p_ctx->p_link);
84  } while (!p_link);
85  ABTI_ythread_set_ready(ABTI_xstream_get_local(p_local_xstream),
86  ABTI_ythread_context_get_ythread(p_link));
87  }
88  }
89 
90  /* No other ULT is waiting or blocked for this ULT. Since a context does not
91  * switch to another context when it finishes, we need to explicitly switch
92  * to the parent. */
93  ABTI_ythread_finish_context_to_parent(p_local_xstream, p_ythread);
94 }

110 void ABTD_ythread_cancel(ABTI_xstream *p_local_xstream, ABTI_ythread *p_ythread)
111 {
112  /* When we cancel a ULT, if other ULT is blocked to join the canceled ULT,
113  * we have to wake up the joiner ULT. However, unlike the case when the
114  * ULT has finished its execution and calls ABTD_ythread_terminate/exit,
115  * this function is called by the scheduler. Therefore, we should not
116  * context switch to the joiner ULT and need to always wake it up. */
117  ABTD_ythread_context *p_ctx = &p_ythread->ctx;
118 
119  if (ABTD_atomic_acquire_load_ythread_context_ptr(&p_ctx->p_link)) {
120  /* If p_link is set, it means that other ULT has called the join. */
121  ABTI_ythread *p_joiner = ABTI_ythread_context_get_ythread(
122  ABTD_atomic_relaxed_load_ythread_context_ptr(&p_ctx->p_link));
123  ABTI_ythread_set_ready(ABTI_xstream_get_local(p_local_xstream),
124  p_joiner);
125  } else {
126  uint32_t req =
127  ABTD_atomic_fetch_or_uint32(&p_ythread->thread.request,
128  ABTI_THREAD_REQ_JOIN |
129  ABTI_THREAD_REQ_TERMINATE);
130  if (req & ABTI_THREAD_REQ_JOIN) {
131  /* This case means there has been a join request and the joiner has
132  * blocked. We have to wake up the joiner ULT. */
133  while (ABTD_atomic_acquire_load_ythread_context_ptr(
134  &p_ctx->p_link) == NULL)
135  ;
136  ABTI_ythread *p_joiner = ABTI_ythread_context_get_ythread(
137  ABTD_atomic_relaxed_load_ythread_context_ptr(&p_ctx->p_link));
138  ABTI_ythread_set_ready(ABTI_xstream_get_local(p_local_xstream),
139  p_joiner);
140  }
141  }
142  ABTI_tool_event_thread_cancel(p_local_xstream, &p_ythread->thread);
143 }
