85 static void sched_run(ABT_sched sched)
86 {
87  ABTI_global *p_global = ABTI_global_get_global();
88  ABTI_xstream *p_local_xstream =
89  ABTI_local_get_xstream(ABTI_local_get_local());
90  uint32_t work_count = 0;
91  sched_data *p_data;
92  uint32_t event_freq;
93  int num_pools;
94  ABT_pool *pools;
95  int i;
96  int run_cnt_nowait;
97 
98  ABTI_sched *p_sched = ABTI_sched_get_ptr(sched);
99  ABTI_ASSERT(p_sched);
100 
101  p_data = sched_data_get_ptr(p_sched->data);
102  event_freq = p_data->event_freq;
103  num_pools = p_data->num_pools;
104  pools = p_data->pools;
105 
106  while (1) {
107  run_cnt_nowait = 0;
108 
109  /* Execute one work unit from the scheduler's pool */
110  for (i = 0; i < num_pools; i++) {
111  ABT_pool pool = pools[i];
112  ABTI_pool *p_pool = ABTI_pool_get_ptr(pool);
113  /* Pop one work unit */
114  ABT_unit unit = ABTI_pool_pop(p_pool);
115  if (unit != ABT_UNIT_NULL) {
116  ABTI_xstream_run_unit(p_global, &p_local_xstream, unit, p_pool);
117  run_cnt_nowait++;
118  break;
119  }
120  }
121 
122  /* Block briefly on pop_wait() if we didn't find work to do in main loop
123  * above. */
124  if (!run_cnt_nowait) {
125  ABTI_pool *p_pool = ABTI_pool_get_ptr(pools[0]);
126  ABT_unit unit;
127  if (p_pool->p_pop_wait) {
128  unit = ABTI_pool_pop_wait(p_pool, 0.1);
129  } else if (p_pool->p_pop_timedwait) {
130  unit = ABTI_pool_pop_timedwait(p_pool, ABTI_get_wtime() + 0.1);
131  } else {
132  /* No "wait" pop, so let's use a normal one. */
133  unit = ABTI_pool_pop(p_pool);
134  }
135  if (unit != ABT_UNIT_NULL) {
136  ABTI_xstream_run_unit(p_global, &p_local_xstream, unit,
137  ABTI_pool_get_ptr(pools[0]));
138  break;
139  }
140  }
141 
142  /* If run_cnt_nowait is zero, that means that no units were found in
143  * first pass through pools and we must have called pop_wait above. We
144  * should check events regardless of work_count in that case for them to
145  * be processed in a timely manner. */
146  if (!run_cnt_nowait || (++work_count >= event_freq)) {
147  ABTI_xstream_check_events(p_local_xstream, p_sched);
148  ABTI_local *p_local = ABTI_xstream_get_local(p_local_xstream);
149  if (ABTI_sched_has_to_stop(&p_local, p_sched) == ABT_TRUE)
150  break;
151  p_local_xstream = ABTI_local_get_xstream(p_local);
152  work_count = 0;
153  }
154  }
155 }
