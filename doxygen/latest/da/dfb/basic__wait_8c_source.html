91 static void sched_run(ABT_sched sched)
92 {
93  ABTI_xstream *p_local_xstream =
94  ABTI_local_get_xstream(ABTI_local_get_local());
95  uint32_t work_count = 0;
96  sched_data *p_data;
97  uint32_t event_freq;
98  int num_pools;
99  ABT_pool *pools;
100  int i;
101  int run_cnt_nowait;
102 
103  ABTI_sched *p_sched = ABTI_sched_get_ptr(sched);
104  ABTI_ASSERT(p_sched);
105 
106  p_data = sched_data_get_ptr(p_sched->data);
107  event_freq = p_data->event_freq;
108  num_pools = p_data->num_pools;
109  pools = p_data->pools;
110 
111  while (1) {
112  run_cnt_nowait = 0;
113 
114  /* Execute one work unit from the scheduler's pool */
115  for (i = 0; i < num_pools; i++) {
116  ABT_pool pool = pools[i];
117  ABTI_pool *p_pool = ABTI_pool_get_ptr(pool);
118  /* Pop one work unit */
119  ABT_unit unit = ABTI_pool_pop(p_pool);
120  if (unit != ABT_UNIT_NULL) {
121  ABTI_xstream_run_unit(&p_local_xstream, unit, p_pool);
122  run_cnt_nowait++;
123  break;
124  }
125  }
126 
127  /* Block briefly on pop_wait() if we didn't find work to do in main loop
128  * above. */
129  if (!run_cnt_nowait) {
130  ABT_unit unit =
131  ABTI_pool_pop_wait(ABTI_pool_get_ptr(pools[0]), 0.1);
132  if (unit != ABT_UNIT_NULL) {
133  ABTI_xstream_run_unit(&p_local_xstream, unit,
134  ABTI_pool_get_ptr(pools[0]));
135  break;
136  }
137  }
138 
139  /* If run_cnt_nowait is zero, that means that no units were found in
140  * first pass through pools and we must have called pop_wait above. We
141  * should check events regardless of work_count in that case for them to
142  * be processed in a timely manner. */
143  if (!run_cnt_nowait || (++work_count >= event_freq)) {
144  ABTI_xstream_check_events(p_local_xstream, p_sched);
145  ABTI_local *p_local = ABTI_xstream_get_local(p_local_xstream);
146  if (ABTI_sched_has_to_stop(&p_local, p_sched) == ABT_TRUE)
147  break;
148  p_local_xstream = ABTI_local_get_xstream(p_local);
149  work_count = 0;
150  }
151  }
152 }
