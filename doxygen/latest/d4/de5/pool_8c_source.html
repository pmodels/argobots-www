476 int ABT_pool_add_sched(ABT_pool pool, ABT_sched sched)
477 {
478  int abt_errno = ABT_SUCCESS;
479  ABTI_xstream *p_local_xstream = ABTI_local_get_xstream();
480 
481  ABTI_pool *p_pool = ABTI_pool_get_ptr(pool);
482  ABTI_CHECK_NULL_POOL_PTR(p_pool);
483 
484  ABTI_sched *p_sched = ABTI_sched_get_ptr(sched);
485  ABTI_CHECK_NULL_SCHED_PTR(p_sched);
486 
487 #ifndef ABT_CONFIG_DISABLE_POOL_CONSUMER_CHECK
488  int p;
489 
490  switch (p_pool->access) {
491  case ABT_POOL_ACCESS_PRIV:
492  case ABT_POOL_ACCESS_SPSC:
493  case ABT_POOL_ACCESS_MPSC:
494  /* we need to ensure that the target pool has already an
495  * associated ES */
496  ABTI_CHECK_TRUE(p_pool->consumer_id != 0, ABT_ERR_POOL);
497 
498  /* We check that from the pool set of the scheduler we do not find
499  * a pool with another associated pool, and set the right value if
500  * it is okay */
501  for (p = 0; p < p_sched->num_pools; p++) {
502  abt_errno =
503  ABTI_pool_set_consumer(ABTI_pool_get_ptr(p_sched->pools[p]),
504  p_pool->consumer_id);
505  ABTI_CHECK_ERROR(abt_errno);
506  }
507  break;
508 
509  case ABT_POOL_ACCESS_SPMC:
510  case ABT_POOL_ACCESS_MPMC:
511  /* we need to ensure that the pool set of the scheduler does
512  * not contain an ES private pool */
513  for (p = 0; p < p_sched->num_pools; p++) {
514  ABTI_pool *p_local_xstream_pool =
515  ABTI_pool_get_ptr(p_sched->pools[p]);
516  ABTI_CHECK_TRUE(p_local_xstream_pool->access !=
517  ABT_POOL_ACCESS_PRIV &&
518  p_local_xstream_pool->access !=
519  ABT_POOL_ACCESS_SPSC &&
520  p_local_xstream_pool->access !=
521  ABT_POOL_ACCESS_MPSC,
522  ABT_ERR_POOL);
523  }
524  break;
525 
526  default:
527  ABTI_CHECK_TRUE(0, ABT_ERR_INV_POOL_ACCESS);
528  }
529 #endif
530 
531  /* Mark the scheduler as it is used in pool */
532  ABTI_CHECK_TRUE(p_sched->used == ABTI_SCHED_NOT_USED, ABT_ERR_INV_SCHED);
533  p_sched->used = ABTI_SCHED_IN_POOL;
