148 int ABT_finalize(void)
149 {
150  int abt_errno = ABT_SUCCESS;
151  ABTI_xstream *p_local_xstream = ABTI_local_get_xstream();
152 
153  /* First, take a global lock protecting the initialization/finalization
154  * process. Don't go to fn_exit before taking a lock */
155  ABTI_spinlock_acquire(&g_ABTI_init_lock);
156 
157  /* If Argobots is not initialized, just return */
158  if (g_ABTI_num_inits == 0) {
159  abt_errno = ABT_ERR_UNINITIALIZED;
160  goto fn_exit;
161  }
162  /* If Argobots is still referenced by others, just return */
163  if (--g_ABTI_num_inits != 0)
164  goto fn_exit;
165 
166  /* If called by an external thread, return an error. */
167  ABTI_CHECK_TRUE(p_local_xstream != NULL, ABT_ERR_INV_XSTREAM);
168 
169  ABTI_CHECK_TRUE_MSG(p_local_xstream->type == ABTI_XSTREAM_TYPE_PRIMARY,
170  ABT_ERR_INV_XSTREAM,
171  "ABT_finalize must be called by the primary ES.");
172 
173  ABTI_unit *p_self = p_local_xstream->p_unit;
174  ABTI_CHECK_TRUE_MSG(p_self->type == ABTI_UNIT_TYPE_THREAD_MAIN,
175  ABT_ERR_INV_THREAD,
176  "ABT_finalize must be called by the primary ULT.");
177  ABTI_thread *p_thread = ABTI_unit_get_thread(p_self);
178 
179 #ifndef ABT_CONFIG_DISABLE_TOOL_INTERFACE
180  /* Turns off the tool interface */
181  ABTI_tool_event_thread_update_callback(NULL, ABT_TOOL_EVENT_THREAD_NONE,
182  NULL);
183  ABTI_tool_event_task_update_callback(NULL, ABT_TOOL_EVENT_TASK_NONE, NULL);
184 #endif
185 
186  /* Set the join request */
187  ABTI_xstream_set_request(p_local_xstream, ABTI_XSTREAM_REQ_JOIN);
188 
189  /* We wait for the remaining jobs */
190  if (ABTD_atomic_acquire_load_int(&p_local_xstream->state) !=
191  ABT_XSTREAM_STATE_TERMINATED) {
192  /* Set the orphan request for the primary ULT */
193  ABTI_thread_set_request(p_thread, ABTI_UNIT_REQ_ORPHAN);
194 
195  LOG_DEBUG("[U%" PRIu64 ":E%d] yield to scheduler\n",
196  ABTI_thread_get_id(p_thread),
197  p_thread->unit_def.p_last_xstream->rank);
198 
199  /* Switch to the parent */
200  ABTI_thread_context_switch_to_parent(&p_local_xstream, p_thread,
201  ABT_SYNC_EVENT_TYPE_XSTREAM_JOIN,
202  (void *)p_local_xstream);
203 
204  /* Back to the original thread */
205  LOG_DEBUG("[U%" PRIu64 ":E%d] resume after yield\n",
206  ABTI_thread_get_id(p_thread),
207  p_thread->unit_def.p_last_xstream->rank);
208  }
209 
210  /* Remove the primary ULT */
211  ABTI_ASSERT(p_local_xstream->p_unit == p_self);
212  p_local_xstream->p_unit = NULL;
213  ABTI_thread_free_main(p_local_xstream, p_thread);
214 
215  /* Free the primary ES */
216  abt_errno = ABTI_xstream_free(p_local_xstream, p_local_xstream, ABT_TRUE);
217  ABTI_CHECK_ERROR(abt_errno);
218 
219  /* Finalize the ES local data */
220  ABTI_local_set_xstream(NULL);
221

274 void ABTI_global_update_max_xstreams(int new_size)
275 {
276  int i, cur_size;
277 
278  if (new_size != 0 && new_size < gp_ABTI_global->max_xstreams)
279  return;
280 
281  ABTI_spinlock_acquire(&gp_ABTI_global->xstreams_lock);
282 
283  static int max_xstreams_warning_once = 0;
284  if (max_xstreams_warning_once == 0) {
285  /* Because some Argobots functionalities depend on the runtime value
286  * ABT_MAX_NUM_XSTREAMS (or gp_ABTI_global->max_xstreams), changing this
287  * value at run-time can cause an error. For example, using ABT_mutex
288  * created before updating max_xstreams causes an error since
289  * ABTI_thread_htable's array size depends on ABT_MAX_NUM_XSTREAMS.
290  * To fix this issue, please set a larger number to ABT_MAX_NUM_XSTREAMS
291  * in advance. */
292  char *warning_message = (char *)malloc(sizeof(char) * 1024);
293  snprintf(warning_message, 1024,
294  "Warning: the number of execution streams exceeds "
295  "ABT_MAX_NUM_XSTREAMS (=%d), which may cause an unexpected "
296  "error.",
297  gp_ABTI_global->max_xstreams);
298  HANDLE_WARNING(warning_message);
299  free(warning_message);
300  max_xstreams_warning_once = 1;
301  }
302 
303  cur_size = gp_ABTI_global->max_xstreams;
304  new_size = (new_size > 0) ? new_size : cur_size * 2;
305  gp_ABTI_global->max_xstreams = new_size;
306  gp_ABTI_global->p_xstreams =
307  (ABTI_xstream **)ABTU_realloc(gp_ABTI_global->p_xstreams,
308  cur_size * sizeof(ABTI_xstream *),
309  new_size * sizeof(ABTI_xstream *));
310 
311  for (i = gp_ABTI_global->num_xstreams; i < new_size; i++) {
312  gp_ABTI_global->p_xstreams[i] = NULL;
313  }
314 
315  ABTI_spinlock_release(&gp_ABTI_global->xstreams_lock);
316 }
