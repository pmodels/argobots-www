249 ABTU_ret_err static int finailze_library(void)
250 {
251  ABTI_local *p_local = ABTI_local_get_local();
252 
253  /* If Argobots is not initialized, just return */
254  ABTI_CHECK_TRUE(g_ABTI_num_inits > 0, ABT_ERR_UNINITIALIZED);
255  /* If Argobots is still referenced by others, just return */
256  if (--g_ABTI_num_inits != 0) {
257  return ABT_SUCCESS;
258  }
259 
260  ABTI_global *p_global = ABTI_global_get_global();
261  ABTI_xstream *p_local_xstream = ABTI_local_get_xstream_or_null(p_local);
262  /* If called by an external thread, return an error. */
263  ABTI_CHECK_TRUE(!ABTI_IS_EXT_THREAD_ENABLED || p_local_xstream,
264  ABT_ERR_INV_XSTREAM);
265 
266  ABTI_CHECK_TRUE_MSG(p_local_xstream->type == ABTI_XSTREAM_TYPE_PRIMARY,
267  ABT_ERR_INV_XSTREAM,
268  "ABT_finalize must be called by the primary ES.");
269 
270  ABTI_thread *p_self = p_local_xstream->p_thread;
271  ABTI_CHECK_TRUE_MSG(p_self->type & ABTI_THREAD_TYPE_PRIMARY,
272  ABT_ERR_INV_THREAD,
273  "ABT_finalize must be called by the primary ULT.");
274  ABTI_ythread *p_ythread;
275  ABTI_CHECK_YIELDABLE(p_self, &p_ythread, ABT_ERR_INV_THREAD);
276 
277 #ifndef ABT_CONFIG_DISABLE_TOOL_INTERFACE
278  /* Turns off the tool interface */
279  ABTI_tool_event_thread_update_callback(p_global, NULL,
280  ABT_TOOL_EVENT_THREAD_NONE, NULL);
281 #endif
282 
283  /* Set the orphan request for the primary ULT */
284  ABTI_thread_set_request(p_self, ABTI_THREAD_REQ_ORPHAN);
285  /* Finish the main scheduler of this local xstream. */
286  ABTI_sched_finish(p_local_xstream->p_main_sched);
287  /* p_self cannot join the main scheduler since p_self needs to be orphaned.
288  * Let's wait till the main scheduler finishes. This thread will be
289  * scheduled when the main root thread finishes. */
290  ABTI_ythread_yield(&p_local_xstream, p_ythread, ABT_SYNC_EVENT_TYPE_OTHER,
291  NULL);
292  ABTI_ASSERT(p_local_xstream == ABTI_local_get_xstream(p_local));
293  ABTI_ASSERT(p_local_xstream->p_thread == p_self);
294 
295  /* Remove the primary ULT */
296  p_local_xstream->p_thread = NULL;
297  ABTI_ythread_free_primary(p_global, ABTI_xstream_get_local(p_local_xstream),
298  p_ythread);
299 
300  /* Free the primary ES */
301  ABTI_xstream_free(p_global, ABTI_xstream_get_local(p_local_xstream),
302  p_local_xstream, ABT_TRUE);
303

305  ABTI_local_set_xstream(NULL);
306

309 
310  /* Finalize the memory pool */
311  ABTI_mem_finalize(p_global);
312

314  if (p_global->set_affinity == ABT_TRUE) {
315  ABTD_affinity_finalize(p_global);
316  }
317

319  ABTU_free(p_global);
320  ABTI_global_set_global(NULL);
321  ABTD_atomic_release_store_uint32(&g_ABTI_initialized, 0);
