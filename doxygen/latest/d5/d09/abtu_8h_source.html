75 static inline uint32_t ABTU_roundup_uint32(uint32_t val, uint32_t multiple)
76 {
77  if ((multiple & (multiple - 1)) == 0) {
78  /* If multiple is a power of two. */
79  return (val + multiple - 1) & (~(multiple - 1));
80  } else {
