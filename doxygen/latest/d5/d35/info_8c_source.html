879 #define PRINT_STACK_FLAG_INITIALIZE 1
880 #define PRINT_STACK_FLAG_WAIT 2
881 #define PRINT_STACK_FLAG_FINALIZE 3

883 static ABTD_atomic_int print_stack_flag =
884  ABTD_ATOMIC_INT_STATIC_INITIALIZER(PRINT_STACK_FLAG_UNSET);
885 static FILE *print_stack_fp = NULL;
886 static double print_stack_timeout = 0.0;
887 static void (*print_cb_func)(ABT_bool, void *) = NULL;
888 static void *print_arg = NULL;
889 static ABTD_atomic_int print_stack_barrier =
890  ABTD_ATOMIC_INT_STATIC_INITIALIZER(0);

892 void ABTI_info_check_print_all_thread_stacks(void)
893 {
894  if (ABTD_atomic_acquire_load_int(&print_stack_flag) !=
895  PRINT_STACK_FLAG_WAIT)
896  return;
897 
898  /* Wait for the other execution streams using a barrier mechanism. */
899  int self_value = ABTD_atomic_fetch_add_int(&print_stack_barrier, 1);
900  if (self_value == 0) {
901  /* This ES becomes the main ES. */
902  double start_time = ABTI_get_wtime();
903  ABT_bool force_print = ABT_FALSE;
904 
905  /* xstreams_lock is acquired to avoid dynamic ES creation while
906  * printing data. */
907  ABTI_spinlock_acquire(&gp_ABTI_global->xstream_list_lock);
908  while (1) {
909  if (ABTD_atomic_acquire_load_int(&print_stack_barrier) >=
910  gp_ABTI_global->num_xstreams) {
911  break;
912  }
913  if (print_stack_timeout >= 0.0 &&
914  (ABTI_get_wtime() - start_time) >= print_stack_timeout) {
915  force_print = ABT_TRUE;
916  break;
917  }
918  ABTI_spinlock_release(&gp_ABTI_global->xstream_list_lock);
919  ABTD_atomic_pause();
920  ABTI_spinlock_acquire(&gp_ABTI_global->xstream_list_lock);
921  }
922  /* All the available ESs are (supposed to be) stopped. We *assume* that
923  * no ES is calling and will call Argobots functions except this
924  * function while printing stack information. */
925  if (force_print) {
926  fprintf(print_stack_fp,
927  "ABT_info_trigger_print_all_thread_stacks: "
928  "timeout (only %d ESs stop)\n",
929  ABTD_atomic_acquire_load_int(&print_stack_barrier));
930  }
931  int abt_errno = print_all_thread_stacks(print_stack_fp);
932  if (ABTI_IS_ERROR_CHECK_ENABLED && abt_errno != ABT_SUCCESS) {
933  fprintf(print_stack_fp, "ABT_info_trigger_print_all_thread_stacks: "
934  "failed because of an internal error.\n");
935  }
936  fflush(print_stack_fp);
937  /* Release the lock that protects ES data. */
938  ABTI_spinlock_release(&gp_ABTI_global->xstream_list_lock);
939  if (print_cb_func)
940  print_cb_func(force_print, print_arg);
941  /* Update print_stack_flag to 3. */
942  ABTD_atomic_release_store_int(&print_stack_flag,
943  PRINT_STACK_FLAG_FINALIZE);
944  } else {
945  /* Wait for the main ES's work. */
946  while (ABTD_atomic_acquire_load_int(&print_stack_flag) !=
947  PRINT_STACK_FLAG_FINALIZE)
948  ABTD_atomic_pause();
949  }
950  ABTI_ASSERT(ABTD_atomic_acquire_load_int(&print_stack_flag) ==
951  PRINT_STACK_FLAG_FINALIZE);
952 
953  /* Decrement the barrier value. */
954  int dec_value = ABTD_atomic_fetch_sub_int(&print_stack_barrier, 1);
955  if (dec_value == 0) {
956  /* The last execution stream resets the flag. */
957  ABTD_atomic_release_store_int(&print_stack_flag,
958  PRINT_STACK_FLAG_UNSET);
959  }
960 }
