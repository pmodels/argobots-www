77 /* Set the associated consumer ES of a pool. This function has no effect on
78  * pools of shared-read access mode. If a pool is private-read to an ES, we
79  * check that the previous value of "consumer_id" is the same as the argument of
80  * the function "consumer_id"

82 #ifndef ABT_CONFIG_DISABLE_POOL_CONSUMER_CHECK
83 static inline int
84 ABTI_pool_set_consumer_common_impl(ABTI_pool *p_pool,
85  ABTI_native_thread_id consumer_id)
86 {
87  switch (p_pool->access) {
88  case ABT_POOL_ACCESS_PRIV:
89 #ifndef ABT_CONFIG_DISABLE_POOL_PRODUCER_CHECK
90  ABTI_CHECK_TRUE_RET(!p_pool->producer_id ||
91  p_pool->producer_id == consumer_id,
92  ABT_ERR_INV_POOL_ACCESS);
93 #endif
94  ABTI_CHECK_TRUE_RET(!p_pool->consumer_id ||
95  p_pool->consumer_id == consumer_id,
96  ABT_ERR_INV_POOL_ACCESS);
97  p_pool->consumer_id = consumer_id;
98  break;
99 
100  case ABT_POOL_ACCESS_SPSC:
101  case ABT_POOL_ACCESS_MPSC:
102  ABTI_CHECK_TRUE_RET(!p_pool->consumer_id ||
103  p_pool->consumer_id == consumer_id,
104  ABT_ERR_INV_POOL_ACCESS);
105  /* NB: as we do not want to use a mutex, the function can be wrong
106  * here */
107  p_pool->consumer_id = consumer_id;
108  break;
109 
110  case ABT_POOL_ACCESS_SPMC:
111  case ABT_POOL_ACCESS_MPMC:
112  p_pool->consumer_id = consumer_id;
113  break;
114 
115  default:
116  return ABT_ERR_INV_POOL_ACCESS;
117  }
118  return ABT_SUCCESS;
