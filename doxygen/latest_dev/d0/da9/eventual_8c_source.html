229 int ABT_eventual_set(ABT_eventual eventual, void *value, int nbytes)
230 {
231  int abt_errno = ABT_SUCCESS;
232  ABTI_local *p_local = ABTI_local_get_local();
233  ABTI_eventual *p_eventual = ABTI_eventual_get_ptr(eventual);
234  ABTI_CHECK_NULL_EVENTUAL_PTR(p_eventual);
235  ABTI_CHECK_TRUE(nbytes <= p_eventual->nbytes, ABT_ERR_INV_EVENTUAL);
236 
237  ABTI_spinlock_acquire(&p_eventual->lock);
238 
239  p_eventual->ready = ABT_TRUE;
240  if (p_eventual->value)
241  memcpy(p_eventual->value, value, nbytes);
242 
243  if (p_eventual->p_head == NULL) {
244  ABTI_spinlock_release(&p_eventual->lock);
245  goto fn_exit;
246  }
247 
248  /* Wake up all waiting ULTs */
249  ABTI_thread *p_head = p_eventual->p_head;
250  ABTI_thread *p_thread = p_head;
251  while (1) {
252  ABTI_thread *p_next = p_thread->p_next;
253  p_thread->p_next = NULL;
254 
255  ABTI_ythread *p_ythread = ABTI_thread_get_ythread_or_null(p_thread);
256  if (p_ythread) {
257  ABTI_ythread_set_ready(p_local, p_ythread);
258  } else {
259  /* When the head is an external thread */
260  ABTD_atomic_release_store_int(&p_thread->state,
261  ABTI_THREAD_STATE_READY);
262  }
263 
264  /* Next ULT */
265  if (p_next != NULL) {
266  p_thread = p_next;
267  } else {
268  break;
269  }
270  }
271 
272  p_eventual->p_head = NULL;
273  p_eventual->p_tail = NULL;
274 
275  ABTI_spinlock_release(&p_eventual->lock);
276 
277 fn_exit:
278  return abt_errno;
279
