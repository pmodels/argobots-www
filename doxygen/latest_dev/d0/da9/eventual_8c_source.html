230 int ABT_eventual_set(ABT_eventual eventual, void *value, int nbytes)
231 {
232  int abt_errno = ABT_SUCCESS;
233  ABTI_xstream *p_local_xstream = ABTI_local_get_xstream();
234  ABTI_eventual *p_eventual = ABTI_eventual_get_ptr(eventual);
235  ABTI_CHECK_NULL_EVENTUAL_PTR(p_eventual);
236  ABTI_CHECK_TRUE(nbytes <= p_eventual->nbytes, ABT_ERR_INV_EVENTUAL);
237 
238  ABTI_spinlock_acquire(&p_eventual->lock);
239 
240  p_eventual->ready = ABT_TRUE;
241  if (p_eventual->value)
242  memcpy(p_eventual->value, value, nbytes);
243 
244  if (p_eventual->p_head == NULL) {
245  ABTI_spinlock_release(&p_eventual->lock);
246  goto fn_exit;
247  }
248 
249  /* Wake up all waiting ULTs */
250  ABTI_unit *p_head = p_eventual->p_head;
251  ABTI_unit *p_unit = p_head;
252  while (1) {
253  ABTI_unit *p_next = p_unit->p_next;
254  p_unit->p_next = NULL;
255 
256  if (ABTI_unit_type_is_thread(p_unit->type)) {
257  ABTI_thread *p_thread = ABTI_unit_get_thread(p_unit);
258  ABTI_thread_set_ready(p_local_xstream, p_thread);
259  } else {
260  /* When the head is an external thread */
261  ABTD_atomic_release_store_int(&p_unit->state,
262  ABTI_UNIT_STATE_READY);
263  }
264 
265  /* Next ULT */
266  if (p_next != NULL) {
267  p_unit = p_next;
268  } else {
269  break;
270  }
271  }
272 
273  p_eventual->p_head = NULL;
274  p_eventual->p_tail = NULL;
275 
276  ABTI_spinlock_release(&p_eventual->lock);
277
