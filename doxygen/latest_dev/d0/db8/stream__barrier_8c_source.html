93 int ABT_xstream_barrier_wait(ABT_xstream_barrier barrier)
94 {
95  ABTI_xstream_barrier *p_barrier = ABTI_xstream_barrier_get_ptr(barrier);
96  ABTI_CHECK_NULL_XSTREAM_BARRIER_PTR(p_barrier);
97 
98  if (p_barrier->num_waiters > 1) {
99 #ifdef HAVE_PTHREAD_BARRIER_INIT
100  ABTD_xstream_barrier_wait(&p_barrier->bar);
101 #else
102  /* The following implementation is a simple sense-reversal barrier
103  * implementation while it uses uint64_t instead of boolean to prevent
104  * a sense variable from wrapping around. */
105  ABTI_spinlock_acquire(&p_barrier->lock);
106  p_barrier->counter++;
107  if (p_barrier->counter == p_barrier->num_waiters) {
108  /* Wake up the other waiters. */
109  p_barrier->counter = 0;
110  /* Updating tag wakes up other waiters. Note that this tag is
111  * sufficiently large, so it will not wrap around. */
112  uint64_t cur_tag = ABTD_atomic_relaxed_load_uint64(&p_barrier->tag);
113  uint64_t new_tag = (cur_tag + 1) & (UINT64_MAX >> 1);
114  ABTD_atomic_release_store_uint64(&p_barrier->tag, new_tag);
115  ABTI_spinlock_release(&p_barrier->lock);
116  } else {
117  /* Wait until the tag is updated by the last waiter */
118  uint64_t cur_tag = ABTD_atomic_relaxed_load_uint64(&p_barrier->tag);
119  ABTI_spinlock_release(&p_barrier->lock);
120  while (cur_tag == ABTD_atomic_acquire_load_uint64(&p_barrier->tag))
121  ABTD_atomic_pause();
122  }
123 #endif
124  }
125  return ABT_SUCCESS;
126 }
