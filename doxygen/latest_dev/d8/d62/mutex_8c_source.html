178 {

182  if (ABTI_unit_type_is_thread(type)) {
183  LOG_DEBUG("%p: lock_low - try\n", p_mutex);
184  while (!ABTD_atomic_bool_cas_weak_uint32(&p_mutex->val, 0, 1)) {
185  ABTI_thread_yield(pp_local_xstream,
186  ABTI_unit_get_thread(p_local_xstream->p_unit),

191  } else {

193  }

197  * just call mutex_spinlock. */

214  p_htable,
215  ABT_SYNC_EVENT_TYPE_MUTEX,
216  (void *)p_mutex);
217  if (ret == ABT_TRUE) {
218  /* This ULT became a waiter in the mutex queue */
219  goto check_handover;
220  }
221  }
222 
223  if ((c = ABTD_atomic_val_cas_strong_uint32(&p_mutex->val, 0, 1)) != 0) {
224  if (c != 2) {
225  c = ABTD_atomic_exchange_uint32(&p_mutex->val, 2);
226  }

228  ABTI_mutex_wait_low(pp_local_xstream, p_mutex, 2);
229

233  * state. */

244  p_giver->unit_def.unit,
245  ABTI_self_get_native_thread_id(
246  *pp_local_xstream));
247  break;
248  }
249  }
250 
251  c = ABTD_atomic_exchange_uint32(&p_mutex->val, 2);
252  }
253  }
254  LOG_DEBUG("%p: lock_low - acquired\n", p_mutex);
255  } else {
256  ABTI_mutex_spinlock(p_mutex);
257  }
258 
259 fn_exit:

473 /* Hand over the mutex to other ULT on the same ES */
474 static inline int ABTI_mutex_unlock_se(ABTI_xstream **pp_local_xstream,
475  ABTI_mutex *p_mutex)
476 {
477  int abt_errno = ABT_SUCCESS;
478 
479 #ifdef ABT_CONFIG_USE_SIMPLE_MUTEX
480  ABTD_atomic_release_store_uint32(&p_mutex->val, 0);
481  ABTI_xstream *p_local_xstream = *pp_local_xstream;
482  LOG_DEBUG("%p: unlock_se\n", p_mutex);
483  if (ABTI_unit_type_is_thread(ABTI_self_get_type(p_local_xstream)))
484  ABTI_thread_yield(pp_local_xstream,
485  ABTI_unit_get_thread(p_local_xstream->p_unit),
486  ABT_SYNC_EVENT_TYPE_MUTEX, (void *)p_mutex);
487 #else
488  int i;
489  ABTI_thread *p_next = NULL;
490  ABTI_thread *p_thread;
491  ABTI_xstream *p_local_xstream = *pp_local_xstream;
492  ABTI_thread_queue *p_queue;
493 
494  /* Unlock the mutex */
495  /* If p_mutex->val is 1 before decreasing it, it means there is no any
496  * waiter in the mutex queue. We can just return. */
497  if (ABTD_atomic_fetch_sub_uint32(&p_mutex->val, 1) == 1) {
498  LOG_DEBUG("%p: unlock_se\n", p_mutex);
499  if (ABTI_unit_type_is_thread(ABTI_self_get_type(p_local_xstream)))
500  ABTI_thread_yield(pp_local_xstream,
501  ABTI_unit_get_thread(p_local_xstream->p_unit),
502  ABT_SYNC_EVENT_TYPE_MUTEX, (void *)p_mutex);
503  return abt_errno;
504  }
505 
506  /* There are ULTs waiting in the mutex queue */
507  ABTI_thread_htable *p_htable = p_mutex->p_htable;
508 
509  p_thread = ABTI_unit_get_thread(p_local_xstream->p_unit);
510  i = (int)p_local_xstream->rank;
511  p_queue = &p_htable->queue[i];
512 
513 check_cond:
514  /* Check whether the mutex handover is possible */
515  if (p_queue->num_handovers >= p_mutex->attr.max_handovers) {
516  ABTD_atomic_release_store_uint32(&p_mutex->val, 0); /* Unlock */
517  LOG_DEBUG("%p: unlock_se\n", p_mutex);
518  ABTI_mutex_wake_de(p_local_xstream, p_mutex);
519  p_queue->num_handovers = 0;
520  ABTI_thread_yield(pp_local_xstream, p_thread, ABT_SYNC_EVENT_TYPE_MUTEX,
521  (void *)p_mutex);
522  return abt_errno;
523  }
524 
525  /* Hand over the mutex to high-priority ULTs */
526  if (p_queue->num_threads <= 1) {
527  if (p_htable->h_list != NULL) {
528  ABTD_atomic_release_store_uint32(&p_mutex->val, 0); /* Unlock */
529  LOG_DEBUG("%p: unlock_se\n", p_mutex);
530  ABTI_mutex_wake_de(p_local_xstream, p_mutex);
531  ABTI_thread_yield(pp_local_xstream, p_thread,
532  ABT_SYNC_EVENT_TYPE_MUTEX, (void *)p_mutex);
533  return abt_errno;
534  }
535  } else {
536  p_next = ABTI_thread_htable_pop(p_htable, p_queue);
537  if (p_next == NULL)
538  goto check_cond;
539  else
540  goto handover;
541  }
542 
543  /* When we don't have high-priority ULTs and other ESs don't either,
544  * we hand over the mutex to low-priority ULTs. */
545  if (p_queue->low_num_threads <= 1) {
546  ABTD_atomic_release_store_uint32(&p_mutex->val, 0); /* Unlock */
547  LOG_DEBUG("%p: unlock_se\n", p_mutex);
548  ABTI_mutex_wake_de(p_local_xstream, p_mutex);
549  ABTI_thread_yield(pp_local_xstream, p_thread, ABT_SYNC_EVENT_TYPE_MUTEX,
550  (void *)p_mutex);
551  return abt_errno;
552  } else {
553  p_next = ABTI_thread_htable_pop_low(p_htable, p_queue);
554  if (p_next == NULL)
555  goto check_cond;
556  }
557 
558 handover:
559  /* We don't push p_thread to the pool. Instead, we will yield_to p_thread
560  * directly at the end of this function. */
561  p_queue->num_handovers++;
562 
563  /* We are handing over the mutex */
564  p_mutex->p_handover = p_next;
565  p_mutex->p_giver = p_thread;
566 
567  LOG_DEBUG("%p: handover -> U%" PRIu64 "\n", p_mutex,
568  ABTI_thread_get_id(p_next));
569 
570  /* yield_to the next ULT */
571  while (ABTD_atomic_acquire_load_uint32(&p_next->unit_def.request) &
572  ABTI_UNIT_REQ_BLOCK)
573  ;
574  ABTI_pool_dec_num_blocked(p_next->unit_def.p_pool);
575  ABTD_atomic_release_store_int(&p_next->unit_def.state,
576  ABTI_UNIT_STATE_RUNNING);
577  ABTI_tool_event_thread_resume(p_local_xstream, p_next, &p_thread->unit_def);
578  /* This works as a "yield" for this thread. */
579  ABTI_tool_event_thread_yield(p_local_xstream, p_thread,
580  p_thread->unit_def.p_parent,
581  ABT_SYNC_EVENT_TYPE_MUTEX, (void *)p_mutex);
582  ABTI_thread *p_prev =
583  ABTI_thread_context_switch_to_sibling(pp_local_xstream, p_thread,
584  p_next);
585  /* Invoke an event of thread resume and run. */
586  p_local_xstream = *pp_local_xstream;
587  ABTI_tool_event_thread_run(p_local_xstream, p_thread, &p_prev->unit_def,
588  p_thread->unit_def.p_parent);
589 #endif
590 
591  return abt_errno;
592 }

688  int val)
689 {

693

695  ABTI_ASSERT(rank < p_htable->num_rows);
696  ABTI_thread_queue *p_queue = &p_htable->queue[rank];
697 
698  ABTI_THREAD_HTABLE_LOCK(p_htable->mutex);
699

701  ABTI_THREAD_HTABLE_UNLOCK(p_htable->mutex);

703  }

706  ABTI_thread_htable_add_h_node(p_htable, p_queue);

714

717

739  }

742  ABTI_thread_htable_add_l_node(p_htable, p_queue);
743  }
744

746  ABTI_thread_set_blocked(p_self);
747 
748  /* Push the current ULT to the queue */
749  ABTI_thread_htable_push_low(p_htable, rank, p_self);
750

752  ABTI_THREAD_HTABLE_UNLOCK(p_htable->mutex);
753

759 void ABTI_mutex_wake_de(ABTI_xstream *p_local_xstream, ABTI_mutex *p_mutex)
760 {
761  int n;
762  ABTI_thread *p_thread;
763  ABTI_thread_htable *p_htable = p_mutex->p_htable;
764  int num = p_mutex->attr.max_wakeups;
765  ABTI_thread_queue *p_start, *p_curr;
766 
767  /* Wake up num ULTs in a round-robin manner */
768  for (n = 0; n < num; n++) {
769  p_thread = NULL;
770 
771  ABTI_THREAD_HTABLE_LOCK(p_htable->mutex);
772 
773  if (ABTD_atomic_acquire_load_uint32(&p_htable->num_elems) == 0) {
774  ABTI_THREAD_HTABLE_UNLOCK(p_htable->mutex);
775  break;
776  }
777 
778  /* Wake up the high-priority ULTs */
779  p_start = p_htable->h_list;
780  for (p_curr = p_start; p_curr;) {
781  p_thread = ABTI_thread_htable_pop(p_htable, p_curr);
782  if (p_curr->num_threads == 0) {
783  ABTI_thread_htable_del_h_head(p_htable);
784  } else {
785  p_htable->h_list = p_curr->p_h_next;
786  }
787  if (p_thread != NULL)
788  goto done;
789  p_curr = p_htable->h_list;
790  if (p_curr == p_start)
791  break;
792  }
793 
794  /* Wake up the low-priority ULTs */
795  p_start = p_htable->l_list;
796  for (p_curr = p_start; p_curr;) {
797  p_thread = ABTI_thread_htable_pop_low(p_htable, p_curr);
798  if (p_curr->low_num_threads == 0) {
799  ABTI_thread_htable_del_l_head(p_htable);
800  } else {
801  p_htable->l_list = p_curr->p_l_next;
802  }
803  if (p_thread != NULL)
804  goto done;
805  p_curr = p_htable->l_list;
806  if (p_curr == p_start)
807  break;
808  }
809 
810  /* Nothing to wake up */
811  ABTI_THREAD_HTABLE_UNLOCK(p_htable->mutex);
812  LOG_DEBUG("%p: nothing to wake up\n", p_mutex);
813  break;
814 
815  done:
816  ABTI_THREAD_HTABLE_UNLOCK(p_htable->mutex);
817 
818  /* Push p_thread to the scheduler's pool */
819  LOG_DEBUG("%p: wake up U%" PRIu64 ":E%d\n", p_mutex,
820  ABTI_thread_get_id(p_thread),
821  ABTI_thread_get_xstream_rank(p_thread));
822  ABTI_thread_set_ready(p_local_xstream, p_thread);
823  }
824 }
