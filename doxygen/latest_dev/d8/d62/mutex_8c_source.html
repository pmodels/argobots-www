568 void ABTI_mutex_wake_de(ABTI_local *p_local, ABTI_mutex *p_mutex)
569 {
570  int n;
571  ABTI_ythread *p_ythread;
572  ABTI_ythread_htable *p_htable = p_mutex->p_htable;
573  int num = p_mutex->attr.max_wakeups;
574  ABTI_ythread_queue *p_start, *p_curr;
575 
576  /* Wake up num ULTs in a round-robin manner */
577  for (n = 0; n < num; n++) {
578  p_ythread = NULL;
579 
580  ABTI_THREAD_HTABLE_LOCK(p_htable->mutex);
581 
582  if (ABTD_atomic_acquire_load_uint32(&p_htable->num_elems) == 0) {
583  ABTI_THREAD_HTABLE_UNLOCK(p_htable->mutex);
584  break;
585  }
586 
587  /* Wake up the high-priority ULTs */
588  p_start = p_htable->h_list;
589  for (p_curr = p_start; p_curr;) {
590  p_ythread = ABTI_ythread_htable_pop(p_htable, p_curr);
591  if (p_curr->num_threads == 0) {
592  ABTI_ythread_htable_del_h_head(p_htable);
593  } else {
594  p_htable->h_list = p_curr->p_h_next;
595  }
596  if (p_ythread != NULL)
597  goto done;
598  p_curr = p_htable->h_list;
599  if (p_curr == p_start)
600  break;
601  }
602 
603  /* Wake up the low-priority ULTs */
604  p_start = p_htable->l_list;
605  for (p_curr = p_start; p_curr;) {
606  p_ythread = ABTI_ythread_htable_pop_low(p_htable, p_curr);
607  if (p_curr->low_num_threads == 0) {
608  ABTI_ythread_htable_del_l_head(p_htable);
609  } else {
610  p_htable->l_list = p_curr->p_l_next;
611  }
612  if (p_ythread != NULL)
613  goto done;
614  p_curr = p_htable->l_list;
615  if (p_curr == p_start)
616  break;
617  }
618 
619  /* Nothing to wake up */
620  ABTI_THREAD_HTABLE_UNLOCK(p_htable->mutex);
621  LOG_DEBUG("%p: nothing to wake up\n", p_mutex);
622  break;
623 
624  done:
625  ABTI_THREAD_HTABLE_UNLOCK(p_htable->mutex);
626 
627  /* Push p_ythread to the scheduler's pool */
628  LOG_DEBUG("%p: wake up U%" PRIu64 ":E%d\n", p_mutex,
629  ABTI_thread_get_id(&p_ythread->thread),
630  p_ythread->thread.p_last_xstream
631  ? p_ythread->thread.p_last_xstream->rank
632  : -1);
633  ABTI_ythread_set_ready(p_local, p_ythread);
634  }
635 }

641 static inline void mutex_lock_low(ABTI_local **pp_local, ABTI_mutex *p_mutex)
642 {
643  ABTI_ythread *p_ythread = NULL;
644  ABTI_xstream *p_local_xstream = ABTI_local_get_xstream_or_null(*pp_local);
645  if (!ABTI_IS_EXT_THREAD_ENABLED || p_local_xstream) {
646  p_ythread = ABTI_thread_get_ythread_or_null(p_local_xstream->p_thread);
647  }
648 #ifdef ABT_CONFIG_USE_SIMPLE_MUTEX
649  if (p_ythread) {
650  LOG_DEBUG("%p: lock_low - try\n", p_mutex);
651  while (!ABTD_atomic_bool_cas_strong_uint32(&p_mutex->val, 0, 1)) {
652  ABTI_ythread_yield(&p_local_xstream, p_ythread,
653  ABT_SYNC_EVENT_TYPE_MUTEX, (void *)p_mutex);
654  *pp_local = ABTI_xstream_get_local(p_local_xstream);
655  }
656  LOG_DEBUG("%p: lock_low - acquired\n", p_mutex);
657  } else {
658  ABTI_mutex_spinlock(p_mutex);
659  }
660 #else
661  /* Only ULTs can yield when the mutex has been locked. For others,
662  * just call mutex_spinlock. */
663  if (p_ythread) {
664  int c;
665  LOG_DEBUG("%p: lock_low - try\n", p_mutex);
666  /* If other ULTs associated with the same ES are waiting on the
667  * low-mutex queue, we give the header ULT a chance to try to get
668  * the mutex by context switching to it. */
669  ABTI_ythread_htable *p_htable = p_mutex->p_htable;
670  ABTI_ythread_queue *p_queue = &p_htable->queue[p_local_xstream->rank];
671  if (p_queue->low_num_threads > 0) {
672  ABT_bool ret =
673  ABTI_ythread_htable_switch_low(&p_local_xstream, p_queue,
674  p_ythread, p_htable,
675  ABT_SYNC_EVENT_TYPE_MUTEX,
676  (void *)p_mutex);
677  *pp_local = ABTI_xstream_get_local(p_local_xstream);
678  if (ret == ABT_TRUE) {
679  /* This ULT became a waiter in the mutex queue */
680  goto check_handover;
681  }
682  }
683 
684  if ((c = ABTD_atomic_val_cas_strong_uint32(&p_mutex->val, 0, 1)) != 0) {
685  if (c != 2) {
686  c = ABTD_atomic_exchange_uint32(&p_mutex->val, 2);
687  }
688  while (c != 0) {
689  ABTI_mutex_wait_low(&p_local_xstream, p_mutex, 2);
690  *pp_local = ABTI_xstream_get_local(p_local_xstream);
691 
692  check_handover:
693  /* If the mutex has been handed over to the current ULT from
694  * other ULT on the same ES, we don't need to change the mutex
695  * state. */
696  if (p_mutex->p_handover) {
697  if (p_ythread == p_mutex->p_handover) {
698  p_mutex->p_handover = NULL;
699  ABTD_atomic_release_store_uint32(&p_mutex->val, 2);
700 
701  /* Push the previous ULT to its pool */
702  ABTI_ythread *p_giver = p_mutex->p_giver;
703  ABTD_atomic_release_store_int(&p_giver->thread.state,
704  ABT_THREAD_STATE_READY);
705  ABTI_pool_push(p_giver->thread.p_pool,
706  p_giver->thread.unit);
707  break;
708  }
709  }
710 
711  c = ABTD_atomic_exchange_uint32(&p_mutex->val, 2);
712  }
713  }
714  LOG_DEBUG("%p: lock_low - acquired\n", p_mutex);
715  } else {
716  ABTI_mutex_spinlock(p_mutex);
717  }
718 
719  return;
720 #endif
721 }

740  /* If it is run on a non-yieldable thread. just unlock it. */
741  if (!p_ythread) {
742  ABTD_atomic_release_store_uint32(&p_mutex->val, 0); /* Unlock */
743  LOG_DEBUG("%p: unlock_se\n", p_mutex);
744  ABTI_mutex_wake_de(*pp_local, p_mutex);
745  return;
746  }
747 
748  /* Unlock the mutex */
749  /* If p_mutex->val is 1 before decreasing it, it means there is no any
750  * waiter in the mutex queue. We can just return. */
751  if (ABTD_atomic_fetch_sub_uint32(&p_mutex->val, 1) == 1) {
752  LOG_DEBUG("%p: unlock_se\n", p_mutex);
753  if (p_ythread) {
754  ABTI_ythread_yield(&p_local_xstream, p_ythread,
755  ABT_SYNC_EVENT_TYPE_MUTEX, (void *)p_mutex);
756  *pp_local = ABTI_xstream_get_local(p_local_xstream);
757  }
758  return;
759  }
760 
761  /* There are ULTs waiting in the mutex queue */
762  ABTI_ythread_htable *p_htable = p_mutex->p_htable;
763  ABTI_ythread *p_next = NULL;
764  ABTI_ythread_queue *p_queue = &p_htable->queue[(int)p_local_xstream->rank];
765 
766 check_cond:
767  /* Check whether the mutex handover is possible */
768  if (p_queue->num_handovers >= p_mutex->attr.max_handovers) {
769  ABTD_atomic_release_store_uint32(&p_mutex->val, 0); /* Unlock */
770  LOG_DEBUG("%p: unlock_se\n", p_mutex);
771  ABTI_mutex_wake_de(*pp_local, p_mutex);
772  p_queue->num_handovers = 0;
773  ABTI_ythread_yield(&p_local_xstream, p_ythread,
774  ABT_SYNC_EVENT_TYPE_MUTEX, (void *)p_mutex);
775  *pp_local = ABTI_xstream_get_local(p_local_xstream);
776  return;
777  }
778 
779  /* Hand over the mutex to high-priority ULTs */
780  if (p_queue->num_threads <= 1) {
781  if (p_htable->h_list != NULL) {
782  ABTD_atomic_release_store_uint32(&p_mutex->val, 0); /* Unlock */
783  LOG_DEBUG("%p: unlock_se\n", p_mutex);
784  ABTI_mutex_wake_de(*pp_local, p_mutex);
785  ABTI_ythread_yield(&p_local_xstream, p_ythread,
786  ABT_SYNC_EVENT_TYPE_MUTEX, (void *)p_mutex);
787  *pp_local = ABTI_xstream_get_local(p_local_xstream);
788  return;
789  }
790  } else {
791  p_next = ABTI_ythread_htable_pop(p_htable, p_queue);
792  if (p_next == NULL)
793  goto check_cond;
794  else
795  goto handover;
796  }
797 
798  /* When we don't have high-priority ULTs and other ESs don't either,
799  * we hand over the mutex to low-priority ULTs. */
800  if (p_queue->low_num_threads <= 1) {
801  ABTD_atomic_release_store_uint32(&p_mutex->val, 0); /* Unlock */
802  LOG_DEBUG("%p: unlock_se\n", p_mutex);
803  ABTI_mutex_wake_de(*pp_local, p_mutex);
804  ABTI_ythread_yield(&p_local_xstream, p_ythread,
805  ABT_SYNC_EVENT_TYPE_MUTEX, (void *)p_mutex);
806  *pp_local = ABTI_xstream_get_local(p_local_xstream);
807  return;
808  } else {
809  p_next = ABTI_ythread_htable_pop_low(p_htable, p_queue);
810  if (p_next == NULL)
811  goto check_cond;
812  }
813 
814 handover:
815  /* We don't push p_ythread to the pool. Instead, we will yield_to p_ythread
816  * directly at the end of this function. */
817  p_queue->num_handovers++;
818 
819  /* We are handing over the mutex */
820  p_mutex->p_handover = p_next;
821  p_mutex->p_giver = p_ythread;
822 
823  LOG_DEBUG("%p: handover -> U%" PRIu64 "\n", p_mutex,
824  ABTI_thread_get_id(&p_next->thread));
825 
826  /* yield_to the next ULT */
827  while (ABTD_atomic_acquire_load_uint32(&p_next->thread.request) &
828  ABTI_THREAD_REQ_BLOCK)
829  ;
830  ABTI_pool_dec_num_blocked(p_next->thread.p_pool);
831  ABTD_atomic_release_store_int(&p_next->thread.state,
832  ABT_THREAD_STATE_RUNNING);
833  ABTI_tool_event_ythread_resume(ABTI_xstream_get_local(p_local_xstream),
834  p_next, &p_ythread->thread);
835  /* This works as a "yield" for this thread. */
836  ABTI_tool_event_ythread_yield(p_local_xstream, p_ythread,
837  p_ythread->thread.p_parent,
838  ABT_SYNC_EVENT_TYPE_MUTEX, (void *)p_mutex);
839  ABTI_ythread *p_prev =
840  ABTI_ythread_context_switch_to_sibling(&p_local_xstream, p_ythread,
841  p_next);
842  /* Invoke an event of thread resume and run. */
843  *pp_local = ABTI_xstream_get_local(p_local_xstream);
844  ABTI_tool_event_thread_run(p_local_xstream, &p_ythread->thread,
845  &p_prev->thread, p_ythread->thread.p_parent);
846 #endif
847 }
