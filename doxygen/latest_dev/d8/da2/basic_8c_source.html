92 static void sched_run(ABT_sched sched)
93 {
94  ABTI_xstream *p_local_xstream =
95  ABTI_local_get_xstream(ABTI_local_get_local());
96  ABT_unit unit = ABT_UNIT_NULL;
97  uint32_t pop_count = 0;
98  sched_data *p_data;
99  uint32_t event_freq;
100  int num_pools;
101  ABT_pool *pools;
102  int i;
103 
104  ABTI_sched *p_sched = ABTI_sched_get_ptr(sched);
105  ABTI_ASSERT(p_sched);
106 
107  p_data = sched_data_get_ptr(p_sched->data);
108  event_freq = p_data->event_freq;
109  num_pools = p_data->num_pools;
110  pools = p_data->pools;
111 
112  while (1) {
113  for (i = 0; i < num_pools; i++) {
114  ABTI_pool *p_pool = ABTI_pool_get_ptr(pools[i]);
115  ++pop_count;
116  if ((unit = ABTI_pool_pop(p_pool)) != ABT_UNIT_NULL) {
117  ABTI_xstream_run_unit(&p_local_xstream, unit, p_pool);
118  break;
119  }
120  }
121  /* if we attempted event_freq pops, check for events */
122  if (pop_count >= event_freq) {
123  ABTI_xstream_check_events(p_local_xstream, p_sched);
124  ABTI_local *p_local = ABTI_xstream_get_local(p_local_xstream);
125  if (ABTI_sched_has_to_stop(&p_local, p_sched) == ABT_TRUE)
126  break;
127  p_local_xstream = ABTI_local_get_xstream(p_local);
128  SCHED_SLEEP(unit != ABT_UNIT_NULL, p_data->sleep_time);
129  pop_count = 0;
130  }
131  }
132 }
