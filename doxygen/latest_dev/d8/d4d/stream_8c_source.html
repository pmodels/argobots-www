1230 int ABTI_xstream_join(ABTI_xstream **pp_local_xstream, ABTI_xstream *p_xstream)
1231 {
1232  int abt_errno = ABT_SUCCESS;
1233  ABTI_xstream *p_local_xstream;
1234  ABTI_thread *p_thread = NULL;
1235  ABT_bool is_blockable = ABT_FALSE;
1236 
1237  ABTI_CHECK_TRUE_MSG(p_xstream->type != ABTI_XSTREAM_TYPE_PRIMARY,
1238  ABT_ERR_INV_XSTREAM,
1239  "The primary ES cannot be joined.");
1240 
1241  /* When the associated pool of the caller ULT has multiple-writer access
1242  * mode, the ULT can be blocked. Otherwise, the access mode, if it is a
1243  * single-writer access mode, may be violated because another ES has to set
1244  * the blocked ULT ready. */
1245  p_local_xstream = *pp_local_xstream;
1246  if (p_local_xstream) {
1247  ABTI_unit *p_self = p_local_xstream->p_unit;
1248  if (ABTI_unit_type_is_thread(p_self->type)) {
1249  p_thread = ABTI_unit_get_thread(p_self);
1250  }
1251  }
1252 
1253  if (p_thread) {
1254  ABT_pool_access access = p_thread->unit_def.p_pool->access;
1255  if (access == ABT_POOL_ACCESS_MPSC || access == ABT_POOL_ACCESS_MPMC) {
1256  is_blockable = ABT_TRUE;
1257  }
1258 
1259  /* The target ES must not be the same as the caller ULT's ES if the
1260  * access mode of the associated pool is not MPMC. */
1261  if (access != ABT_POOL_ACCESS_MPMC) {
1262  ABTI_CHECK_TRUE_MSG(p_xstream != p_local_xstream,
1263  ABT_ERR_INV_XSTREAM,
1264  "The target ES should be different.");
1265  }
1266  }
1267 
1268  if (ABTD_atomic_acquire_load_int(&p_xstream->state) ==
1269  ABT_XSTREAM_STATE_TERMINATED) {
1270  goto fn_join;
1271  }
1272 
1273  /* Wait until the target ES terminates */
1274  if (is_blockable == ABT_TRUE) {
1275  ABTI_POOL_SET_CONSUMER(p_thread->unit_def.p_pool,
1276  ABTI_self_get_native_thread_id(p_local_xstream));
1277 
1278  /* Save the caller ULT to set it ready when the ES is terminated */
1279  p_xstream->p_req_arg = (void *)p_thread;
1280  ABTI_thread_set_blocked(p_thread);
1281 
1282  /* Set the join request */
1283  ABTI_xstream_set_request(p_xstream, ABTI_XSTREAM_REQ_JOIN);
1284 
1285  /* If the caller is a ULT, it is blocked here */
1286  ABTI_thread_suspend(pp_local_xstream, p_thread,
1287  ABT_SYNC_EVENT_TYPE_XSTREAM_JOIN,
1288  (void *)p_xstream);
1289  } else {
1290  /* Set the join request */
1291  ABTI_xstream_set_request(p_xstream, ABTI_XSTREAM_REQ_JOIN);
1292 
1293  while (ABTD_atomic_acquire_load_int(&p_xstream->state) !=
1294  ABT_XSTREAM_STATE_TERMINATED) {
1295 #ifndef ABT_CONFIG_DISABLE_EXT_THREAD
1296  if (!ABTI_unit_type_is_thread(
1297  ABTI_self_get_type(p_local_xstream))) {
1298  ABTD_atomic_pause();
1299  continue;
1300  }
1301 #endif
1302  ABTI_thread_yield(pp_local_xstream, p_thread,
1303  ABT_SYNC_EVENT_TYPE_XSTREAM_JOIN,
1304  (void *)p_xstream);
1305  p_local_xstream = *pp_local_xstream;
1306  }
1307  }
1308 
1309 fn_join:
1310  /* Normal join request */
1311  abt_errno = ABTD_xstream_context_join(&p_xstream->ctx);
1312  ABTI_CHECK_ERROR_MSG(abt_errno, "ABTD_xstream_context_join");
1313 
1314 fn_exit:
1315  return abt_errno;
1316 
1317 fn_fail:
1318  goto fn_exit;
1319 }

1360 /* The main scheduler of each ES executes this routine. */
1361 void ABTI_xstream_schedule(void *p_arg)
1362 {
1363  ABTI_xstream *p_local_xstream = ABTI_local_get_xstream();
1364  ABTI_xstream *p_xstream = (ABTI_xstream *)p_arg;
1365 
1366  ABTI_ASSERT(ABTD_atomic_relaxed_load_int(&p_xstream->state) ==
1367  ABT_XSTREAM_STATE_RUNNING);
1368 
1369  while (1) {
1370  uint32_t request;
1371 
1372  /* Execute the run function of scheduler */
1373  ABTI_sched *p_sched = p_xstream->p_main_sched;
1374  ABTI_ASSERT(p_local_xstream->p_unit == &p_sched->p_thread->unit_def);
1375  LOG_DEBUG("[S%" PRIu64 "] start\n", p_sched->id);
1376  p_sched->run(ABTI_sched_get_handle(p_sched));
1377  LOG_DEBUG("[S%" PRIu64 "] end\n", p_sched->id);
1378 
1379  request = ABTD_atomic_acquire_load_uint32(&p_xstream->request);
1380 
1381  /* If there is an exit or a cancel request, the ES terminates
1382  * regardless of remaining work units. */
1383  if ((request & ABTI_XSTREAM_REQ_EXIT) ||
1384  (request & ABTI_XSTREAM_REQ_CANCEL))
1385  break;
1386 
1387  /* When join is requested, the ES terminates after finishing
1388  * execution of all work units. */
1389  if (request & ABTI_XSTREAM_REQ_JOIN) {
1390  if (ABTI_sched_get_effective_size(p_local_xstream,
1391  p_xstream->p_main_sched) == 0) {
1392  /* If a ULT has been blocked on the join call, we make it ready
1393  */
1394  if (p_xstream->p_req_arg) {
1395  ABTI_thread_set_ready(p_local_xstream,
1396  (ABTI_thread *)p_xstream->p_req_arg);
1397  p_xstream->p_req_arg = NULL;
1398  }
1399  break;
1400  }
1401  }
1402  }
1403 
1404  ABTI_ASSERT(p_local_xstream->p_unit ==
1405  &p_xstream->p_main_sched->p_thread->unit_def);
1406 
1407  /* Set the ES's state as TERMINATED */
1408  ABTD_atomic_release_store_int(&p_xstream->state,
1409  ABT_XSTREAM_STATE_TERMINATED);
1410  LOG_DEBUG("[E%d] terminated\n", p_xstream->rank);
1411 
1412  if (p_xstream->type == ABTI_XSTREAM_TYPE_PRIMARY) {
1413  /* Let us jump back to the main thread. */
1414  ABTI_thread_finish_context_sched_to_main_thread(
1415  p_xstream->p_main_sched);
1416  }
1417 }

1419 int ABTI_xstream_schedule_thread(ABTI_xstream **pp_local_xstream,
1420  ABTI_thread *p_thread)
1421 {
1422  int abt_errno = ABT_SUCCESS;
1423  ABTI_xstream *p_local_xstream = *pp_local_xstream;
1424 
1425 #ifndef ABT_CONFIG_DISABLE_THREAD_CANCEL
1426  if (ABTD_atomic_acquire_load_uint32(&p_thread->unit_def.request) &
1427  ABTI_UNIT_REQ_CANCEL) {
1428  LOG_DEBUG("[U%" PRIu64 ":E%d] canceled\n", ABTI_thread_get_id(p_thread),
1429  p_local_xstream->rank);
1430  ABTD_thread_cancel(p_local_xstream, p_thread);
1431  ABTI_xstream_terminate_thread(p_local_xstream, p_thread);
1432  goto fn_exit;
1433  }
1434 #endif
1435 
1436 #ifndef ABT_CONFIG_DISABLE_MIGRATION
1437  if (ABTD_atomic_acquire_load_uint32(&p_thread->unit_def.request) &
1438  ABTI_UNIT_REQ_MIGRATE) {
1439  abt_errno = ABTI_xstream_migrate_thread(p_local_xstream, p_thread);
1440  ABTI_CHECK_ERROR(abt_errno);
1441  goto fn_exit;
1442  }
1443 #endif
1444 
1445  /* Change the last ES */
1446  p_thread->unit_def.p_last_xstream = p_local_xstream;
1447 
1448  /* Change the ULT state */
1449  ABTD_atomic_release_store_int(&p_thread->unit_def.state,
1450  ABTI_UNIT_STATE_RUNNING);
1451 
1452  /* Switch the context */
1453  LOG_DEBUG("[U%" PRIu64 ":E%d] start running\n",
1454  ABTI_thread_get_id(p_thread), p_local_xstream->rank);
1455 
1456  ABTI_thread *p_self = ABTI_unit_get_thread(p_local_xstream->p_unit);
1457  p_thread =
1458  ABTI_thread_context_switch_to_child(pp_local_xstream, p_self, p_thread);
1459  /* The previous ULT (p_thread) may not be the same as one to which the
1460  * context has been switched. */
1461  /* The scheduler continues from here. */
1462  p_local_xstream = *pp_local_xstream;
1463 
1464  LOG_DEBUG("[U%" PRIu64 ":E%d] stopped\n", ABTI_thread_get_id(p_thread),
1465  p_local_xstream->rank);
1466 
1467  /* Request handling. */
1468  /* We do not need to acquire-load request since all critical requests
1469  * (BLOCK, ORPHAN, STOP, and NOPUSH) are written by p_thread. CANCEL might
1470  * be delayed. */
1471  uint32_t request =
1472  ABTD_atomic_acquire_load_uint32(&p_thread->unit_def.request);
1473  if (request & ABTI_UNIT_REQ_STOP) {
1474  /* The ULT has completed its execution or it called the exit request. */
1475  LOG_DEBUG("[U%" PRIu64 ":E%d] %s\n", ABTI_thread_get_id(p_thread),
1476  p_local_xstream->rank,
1477  (request & ABTI_UNIT_REQ_TERMINATE
