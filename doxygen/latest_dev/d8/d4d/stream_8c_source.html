1174 /* The main scheduler of each ES executes this routine. */
1175 void ABTI_xstream_schedule(void *p_arg)
1176 {
1177  ABTI_local *p_local = ABTI_local_get_local();
1178  ABTI_xstream *p_local_xstream = ABTI_local_get_xstream(p_local);
1179  ABTI_xstream *p_xstream = (ABTI_xstream *)p_arg;
1180 
1181  ABTI_ASSERT(ABTD_atomic_relaxed_load_int(&p_xstream->state) ==
1182  ABT_XSTREAM_STATE_RUNNING);
1183 
1184  while (1) {
1185  uint32_t request;
1186 
1187  /* Execute the run function of scheduler */
1188  ABTI_sched *p_sched = p_xstream->p_main_sched;
1189  ABTI_ASSERT(p_local_xstream->p_thread == &p_sched->p_ythread->thread);
1190  LOG_DEBUG("[S%" PRIu64 "] start\n", p_sched->id);
1191  p_sched->run(ABTI_sched_get_handle(p_sched));
1192  LOG_DEBUG("[S%" PRIu64 "] end\n", p_sched->id);
1193 
1194  request = ABTD_atomic_acquire_load_uint32(&p_xstream->request);
1195 
1196  /* If there is an exit or a cancel request, the ES terminates
1197  * regardless of remaining work units. */
1198  if ((request & ABTI_XSTREAM_REQ_EXIT) ||
1199  (request & ABTI_XSTREAM_REQ_CANCEL))
1200  break;
1201 
1202  /* When join is requested, the ES terminates after finishing
1203  * execution of all work units. */
1204  if (request & ABTI_XSTREAM_REQ_JOIN) {
1205  if (ABTI_sched_get_effective_size(p_local,
1206  p_xstream->p_main_sched) == 0) {
1207  /* If a ULT has been blocked on the join call, we make it ready
1208  */
1209  if (p_xstream->p_req_arg) {
1210  ABTI_ythread_set_ready(p_local, (ABTI_ythread *)
1211  p_xstream->p_req_arg);
1212  p_xstream->p_req_arg = NULL;
1213  }
1214  break;
1215  }
1216  }
1217  }
1218 
1219  ABTI_ASSERT(p_local_xstream->p_thread ==
1220  &p_xstream->p_main_sched->p_ythread->thread);
1221 
1222  /* Set the ES's state as TERMINATED */
1223  ABTD_atomic_release_store_int(&p_xstream->state,
1224  ABT_XSTREAM_STATE_TERMINATED);
1225  LOG_DEBUG("[E%d] terminated\n", p_xstream->rank);
1226 
1227  if (p_xstream->type == ABTI_XSTREAM_TYPE_PRIMARY) {
1228  /* Let us jump back to the main thread. */
1229  ABTI_ythread_finish_context_sched_to_main_thread(
1230  p_xstream->p_main_sched);
1231  }
1232 }

1407 static int xstream_join(ABTI_local **pp_local, ABTI_xstream *p_xstream)
1408 {
1409  ABT_bool is_blockable = ABT_FALSE;
1410 
1411  /* The primary ES cannot be joined. */
1412  ABTI_CHECK_TRUE_RET(p_xstream->type != ABTI_XSTREAM_TYPE_PRIMARY,
1413  ABT_ERR_INV_XSTREAM);
1414 
1415  /* When the associated pool of the caller ULT has multiple-writer access
1416  * mode, the ULT can be blocked. Otherwise, the access mode, if it is a
1417  * single-writer access mode, may be violated because another ES has to set
1418  * the blocked ULT ready. */
1419  ABTI_ythread *p_ythread = NULL;
1420  ABTI_xstream *p_local_xstream = ABTI_local_get_xstream_or_null(*pp_local);
1421  if (!ABTI_IS_EXT_THREAD_ENABLED || p_local_xstream) {
1422  ABTI_thread *p_thread = p_local_xstream->p_thread;
1423  ABT_pool_access access = p_thread->p_pool->access;
1424  /* The target ES must not be the same as the caller ULT's ES if the
1425  * access mode of the associated pool is not MPMC. */
1426  if (access != ABT_POOL_ACCESS_MPMC) {
1427  /* The target ES should be different. */
1428  ABTI_CHECK_TRUE_RET(p_xstream != p_local_xstream,
1429  ABT_ERR_INV_XSTREAM);
1430  }
1431  p_ythread = ABTI_thread_get_ythread_or_null(p_thread);
1432  if (p_ythread) {
1433  if (access == ABT_POOL_ACCESS_MPSC ||
1434  access == ABT_POOL_ACCESS_MPMC) {
1435  is_blockable = ABT_TRUE;
1436  }
1437  }
1438  }
1439 
1440  if (ABTD_atomic_acquire_load_int(&p_xstream->state) !=
1441  ABT_XSTREAM_STATE_TERMINATED) {
1442  /* Wait until the target ES terminates */
1443  if (is_blockable == ABT_TRUE) {
1444  ABTI_pool_set_consumer(ABTI_xstream_get_local(p_local_xstream),
1445  p_ythread->thread.p_pool);
1446 
1447  /* Save the caller ULT to set it ready when the ES is terminated */
1448  p_xstream->p_req_arg = (void *)p_ythread;
1449  ABTI_ythread_set_blocked(p_ythread);
1450 
1451  /* Set the join request */
1452  ABTI_xstream_set_request(p_xstream, ABTI_XSTREAM_REQ_JOIN);
1453 
1454  /* If the caller is a ULT, it is blocked here */
1455  ABTI_ythread_suspend(&p_local_xstream, p_ythread,
1456  ABT_SYNC_EVENT_TYPE_XSTREAM_JOIN,
1457  (void *)p_xstream);
1458  *pp_local = ABTI_xstream_get_local(p_local_xstream);
1459  } else {
1460  /* Set the join request */
1461  ABTI_xstream_set_request(p_xstream, ABTI_XSTREAM_REQ_JOIN);
1462 
1463  while (ABTD_atomic_acquire_load_int(&p_xstream->state) !=
1464  ABT_XSTREAM_STATE_TERMINATED) {
1465  if (p_ythread) {
1466  ABTI_ythread_yield(&p_local_xstream, p_ythread,
1467  ABT_SYNC_EVENT_TYPE_XSTREAM_JOIN,
1468  (void *)p_xstream);
1469  *pp_local = ABTI_xstream_get_local(p_local_xstream);
1470  } else {
1471  ABTD_atomic_pause();
1472  }
1473  }
1474  }
1475  }
