1764 static inline void xstream_schedule_ythread(ABTI_global *p_global,
1765  ABTI_xstream **pp_local_xstream,
1766  ABTI_ythread *p_ythread)
1767 {
1768  ABTI_xstream *p_local_xstream = *pp_local_xstream;
1769 
1770 #ifndef ABT_CONFIG_DISABLE_THREAD_CANCEL
1771  if (ABTD_atomic_acquire_load_uint32(&p_ythread->thread.request) &
1772  ABTI_THREAD_REQ_CANCEL) {
1773  LOG_DEBUG("[U%" PRIu64 ":E%d] canceled\n",
1774  ABTI_thread_get_id(&p_ythread->thread),
1775  p_local_xstream->rank);
1776  ABTD_ythread_cancel(p_local_xstream, p_ythread);
1777  ABTI_xstream_terminate_thread(p_global,
1778  ABTI_xstream_get_local(p_local_xstream),
1779  &p_ythread->thread);
1780  return;
1781  }
1782 #endif
1783 
1784 #ifndef ABT_CONFIG_DISABLE_MIGRATION
1785  if (ABTD_atomic_acquire_load_uint32(&p_ythread->thread.request) &
1786  ABTI_THREAD_REQ_MIGRATE) {
1787  int abt_errno =
1788  xstream_migrate_thread(p_global,
1789  ABTI_xstream_get_local(p_local_xstream),
1790  &p_ythread->thread);
1791  if (!ABTI_IS_ERROR_CHECK_ENABLED || abt_errno == ABT_SUCCESS) {
1792  /* Migration succeeded, so we do not need to schedule p_ythread. */
1793  return;
1794  }
1795  }
1796 #endif
1797 
1798  /* Change the last ES */
1799  p_ythread->thread.p_last_xstream = p_local_xstream;
1800 
1801  /* Change the ULT state */
1802  ABTD_atomic_release_store_int(&p_ythread->thread.state,
1803  ABT_THREAD_STATE_RUNNING);
1804 
1805  /* Switch the context */
1806  LOG_DEBUG("[U%" PRIu64 ":E%d] start running\n",
1807  ABTI_thread_get_id(&p_ythread->thread), p_local_xstream->rank);
1808 
1809  /* Since the argument is pp_local_xstream, p_local_xstream->p_thread must be
1810  * yieldable. */
1811  ABTI_ythread *p_self = ABTI_thread_get_ythread(p_local_xstream->p_thread);
1812  p_ythread = ABTI_ythread_context_switch_to_child(pp_local_xstream, p_self,
1813  p_ythread);
1814  /* The previous ULT (p_ythread) may not be the same as one to which the
1815  * context has been switched. */
1816  /* The scheduler continues from here. */
1817  p_local_xstream = *pp_local_xstream;
1818 
1819  LOG_DEBUG("[U%" PRIu64 ":E%d] stopped\n",
1820  ABTI_thread_get_id(&p_ythread->thread), p_local_xstream->rank);
1821 
1822  /* Request handling. */
1823  /* We do not need to acquire-load request since all critical requests
1824  * (BLOCK, ORPHAN, STOP, and NOPUSH) are written by p_ythread. CANCEL might
1825  * be delayed. */
1826  uint32_t request =
1827  ABTD_atomic_acquire_load_uint32(&p_ythread->thread.request);
1828  if (request & ABTI_THREAD_REQ_TERMINATE) {
1829  /* The ULT has completed its execution or it called the exit request. */
1830  LOG_DEBUG("[U%" PRIu64 ":E%d] finished\n",
1831  ABTI_thread_get_id(&p_ythread->thread),
1832  p_local_xstream->rank);
1833  ABTI_xstream_terminate_thread(p_global,
1834  ABTI_xstream_get_local(p_local_xstream),
1835  &p_ythread->thread);
1836 #ifndef ABT_CONFIG_DISABLE_THREAD_CANCEL
1837  } else if (request & ABTI_THREAD_REQ_CANCEL) {
1838  LOG_DEBUG("[U%" PRIu64 ":E%d] canceled\n",
1839  ABTI_thread_get_id(&p_ythread->thread),
1840  p_local_xstream->rank);
1841  ABTD_ythread_cancel(p_local_xstream, p_ythread);
1842  ABTI_xstream_terminate_thread(p_global,
1843  ABTI_xstream_get_local(p_local_xstream),
1844  &p_ythread->thread);
1845 #endif
1846  } else if (!(request & ABTI_THREAD_REQ_NON_YIELD)) {
1847  /* The ULT did not finish its execution.
1848  * Change the state of current running ULT and
1849  * add it to the pool again. */
1850  ABTI_pool_add_thread(&p_ythread->thread);
1851  } else if (request & ABTI_THREAD_REQ_BLOCK) {
1852  LOG_DEBUG("[U%" PRIu64 ":E%d] check blocked\n",
1853  ABTI_thread_get_id(&p_ythread->thread),
1854  p_local_xstream->rank);
1855  ABTI_thread_unset_request(&p_ythread->thread, ABTI_THREAD_REQ_BLOCK);
1856 #ifndef ABT_CONFIG_DISABLE_MIGRATION
1857  } else if (request & ABTI_THREAD_REQ_MIGRATE) {
1858  /* This is the case when the ULT requests migration of itself. */
1859  int abt_errno =
1860  xstream_migrate_thread(p_global,
1861  ABTI_xstream_get_local(p_local_xstream),
1862  &p_ythread->thread);
1863  /* Migration is optional, so it is okay if it fails. */
1864  (void)abt_errno;
1865 #endif
1866  } else if (request & ABTI_THREAD_REQ_ORPHAN) {
1867  /* The ULT is not pushed back to the pool and is disconnected from any
1868  * pool. */
1869  LOG_DEBUG("[U%" PRIu64 ":E%d] orphaned\n",
1870  ABTI_thread_get_id(&p_ythread->thread),
1871  p_local_xstream->rank);
1872  ABTI_thread_unset_request(&p_ythread->thread, ABTI_THREAD_REQ_ORPHAN);
1873  p_ythread->thread.p_pool->u_free(&p_ythread->thread.unit);
1874  p_ythread->thread.p_pool = NULL;
1875  } else if (request & ABTI_THREAD_REQ_NOPUSH) {
1876  /* The ULT is not pushed back to the pool */
1877  LOG_DEBUG("[U%" PRIu64 ":E%d] not pushed\n",
1878  ABTI_thread_get_id(&p_ythread->thread),
1879  p_local_xstream->rank);
1880  ABTI_thread_unset_request(&p_ythread->thread, ABTI_THREAD_REQ_NOPUSH);
1881  } else {
1882  ABTI_ASSERT(0);
1883  ABTU_unreachable();
1884  }
1885 }

1887 static inline void xstream_schedule_task(ABTI_global *p_global,
1888  ABTI_xstream *p_local_xstream,
1889  ABTI_thread *p_task)
1890 {
1891 #ifndef ABT_CONFIG_DISABLE_TASK_CANCEL
1892  if (ABTD_atomic_acquire_load_uint32(&p_task->request) &
1893  ABTI_THREAD_REQ_CANCEL) {
1894  ABTI_tool_event_thread_cancel(p_local_xstream, p_task);
1895  ABTI_xstream_terminate_thread(p_global,
1896  ABTI_xstream_get_local(p_local_xstream),
1897  p_task);
1898  return;
1899  }
1900 #endif
1901 
1902  /* Change the task state */
1903  ABTD_atomic_release_store_int(&p_task->state, ABT_THREAD_STATE_RUNNING);
1904 
1905  /* Set the associated ES */
1906  p_task->p_last_xstream = p_local_xstream;
1907 
1908  /* Execute the task function */
1909  LOG_DEBUG("[T%" PRIu64 ":E%d] running\n", ABTI_thread_get_id(p_task),
1910  p_local_xstream->rank);
1911

1913  p_local_xstream->p_thread = p_task;
1914  p_task->p_parent = p_sched_thread;
1915 
1916  /* Execute the task function */
1917  ABTI_tool_event_thread_run(p_local_xstream, p_task, p_sched_thread,
1918  p_sched_thread);
1919  LOG_DEBUG("[T%" PRIu64 ":E%d] running\n", ABTI_thread_get_id(p_task),
1920  p_local_xstream->rank);
1921  p_task->f_thread(p_task->p_arg);
1922  ABTI_tool_event_thread_finish(p_local_xstream, p_task, p_sched_thread);
1923  LOG_DEBUG("[T%" PRIu64 ":E%d] stopped\n", ABTI_thread_get_id(p_task),
1924  p_local_xstream->rank);
1925 
1926  /* Set the current running scheduler's thread */
1927  p_local_xstream->p_thread = p_sched_thread;
1928 
1929  /* Terminate the tasklet */
1930  ABTI_xstream_terminate_thread(p_global,
1931  ABTI_xstream_get_local(p_local_xstream),
1932  p_task);
1933 }
