144 int ABT_barrier_wait(ABT_barrier barrier)
145 {
146  int abt_errno = ABT_SUCCESS;
147  ABTI_xstream *p_local_xstream = ABTI_local_get_xstream();
148  ABTI_barrier *p_barrier = ABTI_barrier_get_ptr(barrier);
149  ABTI_CHECK_NULL_BARRIER_PTR(p_barrier);
150  uint32_t pos;
151 
152  ABTI_spinlock_acquire(&p_barrier->lock);
153 
154  ABTI_ASSERT(p_barrier->counter < p_barrier->num_waiters);
155  pos = p_barrier->counter++;
156 
157  /* If we do not have all the waiters yet */
158  if (p_barrier->counter < p_barrier->num_waiters) {
159  ABTI_thread *p_thread;
160  ABT_unit_type type;
161  ABTD_atomic_int32 ext_signal = ABTD_ATOMIC_INT32_STATIC_INITIALIZER(0);
162 
163  if (p_local_xstream != NULL) {
164  ABTI_unit *p_self = p_local_xstream->p_unit;
165  ABTI_CHECK_TRUE(ABTI_unit_type_is_thread(p_self->type),
166  ABT_ERR_BARRIER);
167  p_thread = ABTI_unit_get_thread(p_self);
168  type = ABT_UNIT_TYPE_THREAD;
169  } else {
170  /* external thread */
171  /* Check size if ext_signal can be stored in p_thread. */
172  ABTI_STATIC_ASSERT(sizeof(ext_signal) <= sizeof(p_thread));
173  p_thread = (ABTI_thread *)&ext_signal;
174  type = ABT_UNIT_TYPE_EXT;
175  }
176 
177  /* Keep the waiter's information */
178  p_barrier->waiters[pos] = p_thread;
179  p_barrier->waiter_type[pos] = type;
180 
181  if (type == ABT_UNIT_TYPE_THREAD) {
182  /* Change the ULT's state to BLOCKED */
183  ABTI_thread_set_blocked(p_thread);
184  }
185 
186  ABTI_spinlock_release(&p_barrier->lock);
187 
188  if (type == ABT_UNIT_TYPE_THREAD) {
189  /* Suspend the current ULT */
190  ABTI_thread_suspend(&p_local_xstream, p_thread,
191  ABT_SYNC_EVENT_TYPE_BARRIER, (void *)p_barrier);
192  } else {
193  /* External thread is waiting here polling ext_signal. */
194  /* FIXME: need a better implementation */
195  while (!ABTD_atomic_acquire_load_int32(&ext_signal))
196  ;
197  }
198  } else {
199  /* Signal all the waiting ULTs */
200  int i;
201  for (i = 0; i < p_barrier->num_waiters - 1; i++) {
202  ABTI_thread *p_thread = p_barrier->waiters[i];
203  if (p_barrier->waiter_type[i] == ABT_UNIT_TYPE_THREAD) {
204  ABTI_thread_set_ready(p_local_xstream, p_thread);
205  } else {
206  /* When p_cur is an external thread */
207  ABTD_atomic_int32 *p_ext_signal = (ABTD_atomic_int32 *)p_thread;
208  ABTD_atomic_release_store_int32(p_ext_signal, 1);
209  }
210 
211  p_barrier->waiters[i] = NULL;
212  }
213 
214  /* Reset counter */
215  p_barrier->counter = 0;
216 
217  ABTI_spinlock_release(&p_barrier->lock);
218  }
