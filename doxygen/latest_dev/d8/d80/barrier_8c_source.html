188 {
189  ABTI_local *p_local = ABTI_local_get_local();
190  ABTI_barrier *p_barrier = ABTI_barrier_get_ptr(barrier);
191  ABTI_CHECK_NULL_BARRIER_PTR(p_barrier);
192 
193 #ifndef ABT_CONFIG_ENABLE_VER_20_API
194  /* Calling a barrier on a tasklet is not allowed. */
195  if (ABTI_IS_ERROR_CHECK_ENABLED && p_local) {
196  ABTI_xstream *p_local_xstream = ABTI_local_get_xstream(p_local);
197  ABTI_CHECK_TRUE(p_local_xstream->p_thread->type &
198  ABTI_THREAD_TYPE_YIELDABLE,
199  ABT_ERR_BARRIER);
200  }
201 #endif
202 
203  ABTI_spinlock_acquire(&p_barrier->lock);
204 
205  ABTI_ASSERT(p_barrier->counter < p_barrier->num_waiters);
206  p_barrier->counter++;
207 
208  /* If we do not have all the waiters yet */
209  if (p_barrier->counter < p_barrier->num_waiters) {
210  ABTI_waitlist_wait_and_unlock(&p_local, &p_barrier->waitlist,
211  &p_barrier->lock, ABT_FALSE,
212  ABT_SYNC_EVENT_TYPE_BARRIER,
213  (void *)p_barrier);
214  } else {
215  ABTI_waitlist_broadcast(p_local, &p_barrier->waitlist);
216  /* Reset counter */
217  p_barrier->counter = 0;
218  ABTI_spinlock_release(&p_barrier->lock);
219  }
