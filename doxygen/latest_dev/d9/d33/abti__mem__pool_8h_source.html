114 ABTU_ret_err static inline int
115 ABTI_mem_pool_alloc(ABTI_mem_pool_local_pool *p_local_pool, void **p_mem)
116 {
117  size_t bucket_index = p_local_pool->bucket_index;
118  ABTI_mem_pool_header *cur_bucket = p_local_pool->buckets[bucket_index];
119  size_t num_headers_in_cur_bucket = cur_bucket->bucket_info.num_headers;
120  /* At least one header is available in the current bucket, so it must be
121  * larger than 0. */
122  ABTI_ASSERT(num_headers_in_cur_bucket >= 1);
123  if (num_headers_in_cur_bucket == 1) {
124  /*cur_bucket will be empty after allocation. */
125  if (bucket_index == 0) {
126  /* cur_bucket is the last header in this pool.
127  * Let's get some buckets from the global pool. */
128  size_t i;
129  for (i = 0; i < ABT_MEM_POOL_NUM_TAKE_BUCKETS; i++) {
130  int abt_errno =
131  ABTI_mem_pool_take_bucket(p_local_pool->p_global_pool,
132  &p_local_pool->buckets[i]);
133  if (ABTI_IS_ERROR_CHECK_ENABLED && abt_errno != ABT_SUCCESS) {
134  /* Return buckets that have been already taken. */
135  size_t j;
136  for (j = 0; j < i; j++) {
137  ABTI_mem_pool_return_bucket(p_local_pool->p_global_pool,
138  p_local_pool->buckets[j]);
139  }
140  return abt_errno;
141  }
142  }
143  p_local_pool->bucket_index = ABT_MEM_POOL_NUM_TAKE_BUCKETS - 1;
144  } else {
145  p_local_pool->bucket_index = bucket_index - 1;
146  }
147  /* Now buckets[bucket_index] is full of headers. */
148  } else {
149  /* Let's return the header in the bucket. */
150  ABTI_mem_pool_header *p_next = cur_bucket->p_next;
151  p_next->bucket_info.num_headers = num_headers_in_cur_bucket - 1;
152  p_local_pool->buckets[bucket_index] = p_next;
153  }
154  /* At least one header is available in the current bucket. */
155  *p_mem = (void *)cur_bucket;
156  return ABT_SUCCESS;
157 }

159 static inline void ABTI_mem_pool_free(ABTI_mem_pool_local_pool *p_local_pool,
160  void *mem)
161 {
162  /* At least one header is available in the current bucket. */
163  size_t bucket_index = p_local_pool->bucket_index;
164  ABTI_mem_pool_header *p_freed_header = (ABTI_mem_pool_header *)mem;
165  ABTI_mem_pool_header *cur_bucket = p_local_pool->buckets[bucket_index];
166  if (cur_bucket->bucket_info.num_headers ==
167  p_local_pool->num_headers_per_bucket) {
168  /* cur_bucket is full. */
169  if (++bucket_index == ABT_MEM_POOL_MAX_LOCAL_BUCKETS) {
170  size_t i;
171  /* All buckets are full, so let's return some old buckets. */
172  for (i = 0; i < ABT_MEM_POOL_NUM_RETURN_BUCKETS; i++) {
173  ABTI_mem_pool_return_bucket(p_local_pool->p_global_pool,
174  p_local_pool->buckets[i]);
175  }
176  for (i = ABT_MEM_POOL_NUM_RETURN_BUCKETS;
177  i < ABT_MEM_POOL_MAX_LOCAL_BUCKETS; i++) {
178  p_local_pool->buckets[i - ABT_MEM_POOL_NUM_RETURN_BUCKETS] =
179  p_local_pool->buckets[i];
180  }
181  bucket_index = ABT_MEM_POOL_MAX_LOCAL_BUCKETS -
182  ABT_MEM_POOL_NUM_RETURN_BUCKETS;
183  }
184  p_local_pool->bucket_index = bucket_index;
185  p_freed_header->p_next = NULL;
186  p_freed_header->bucket_info.num_headers = 1;
187  } else {
188  p_freed_header->p_next = cur_bucket;
189  p_freed_header->bucket_info.num_headers =
190  cur_bucket->bucket_info.num_headers + 1;
191  }
192  p_local_pool->buckets[bucket_index] = p_freed_header;
193  /* At least one header is available in the current bucket. */
194 }
