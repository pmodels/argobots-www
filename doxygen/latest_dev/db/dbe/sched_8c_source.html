673  ABT_pool *pool_list;

678

681  if (pools[p] == ABT_POOL_NULL) {

685  ABT_TRUE, &p_newpool);

688  /* Remove pools that are already created. */
689  int i;

691  if (pools[i] != ABT_POOL_NULL)

702  }
703  }
704  } else {
705  /* TODO: Check if it works. */

707  }
708

710  switch (predef) {
711  case ABT_SCHED_DEFAULT:
712  case ABT_SCHED_BASIC:
713  abt_errno = sched_create(ABTI_sched_get_basic_def(), num_pools,
714  pool_list, p_config, def_automatic,
715  pp_newsched);
716  break;

728  abt_errno = sched_create(ABTI_sched_get_randws_def(), num_pools,
729  pool_list, p_config, def_automatic,
730  pp_newsched);
731  break;
732  default:
733  abt_errno = ABT_ERR_INV_SCHED_PREDEF;
734  break;
735  }
736  if (ABTI_IS_ERROR_CHECK_ENABLED && abt_errno != ABT_SUCCESS) {
737  /* Remove pools that are already created. */
738  int i;
739  for (i = 0; i < num_pools; i++) {
740  if (pools[i] != ABT_POOL_NULL)
741  continue; /* User given pool. */
742  /* Free a pool created in this function. */
743  ABTI_pool_free(ABTI_pool_get_ptr(pool_list[i]));

800  case ABT_SCHED_BASIC:

839 {

921 /* Compared to \c ABTI_sched_get_total_size, ABTI_sched_get_effective_size does
922  * not count the number of blocked ULTs if a pool has more than one consumer or
923  * the caller ES is not the latest consumer. This is necessary when the ES
924  * associated with the target scheduler has to be joined and the pool is shared
925  * between different schedulers associated with different ESs. */
926 size_t ABTI_sched_get_effective_size(ABTI_local *p_local, ABTI_sched *p_sched)
927 {
928  size_t pool_size = 0, p;
929 
930  for (p = 0; p < p_sched->num_pools; p++) {
931  ABT_pool pool = p_sched->pools[p];
932  ABTI_pool *p_pool = ABTI_pool_get_ptr(pool);
933  pool_size += ABTI_pool_get_size(p_pool);
934  switch (p_pool->access) {
935  case ABT_POOL_ACCESS_PRIV:
936  pool_size +=
937  ABTD_atomic_acquire_load_int32(&p_pool->num_blocked);
938  break;
939  case ABT_POOL_ACCESS_SPSC:
940  case ABT_POOL_ACCESS_MPSC:
941  case ABT_POOL_ACCESS_SPMC:
942  case ABT_POOL_ACCESS_MPMC:
943  if (ABTD_atomic_acquire_load_int32(&p_pool->num_scheds) == 1) {
944  pool_size +=
945  ABTD_atomic_acquire_load_int32(&p_pool->num_blocked);
946  }
947  break;
948  default:
949  break;
950  }
951  }
952 
953  return pool_size;
