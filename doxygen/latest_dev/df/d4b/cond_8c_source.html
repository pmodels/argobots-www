120 static inline void remove_thread(ABTI_cond *p_cond, ABTI_thread *p_thread)
121 {
122  if (p_thread->p_next == NULL)
123  return;
124 
125  ABTI_spinlock_acquire(&p_cond->lock);
126 
127  if (p_thread->p_next == NULL) {
128  ABTI_spinlock_release(&p_cond->lock);
129  return;
130  }
131 
132  /* If p_thread is still in the queue, we have to remove it. */
133  p_cond->num_waiters--;
134  if (p_cond->num_waiters == 0) {
135  p_cond->p_waiter_mutex = NULL;
136  p_cond->p_head = NULL;
137  p_cond->p_tail = NULL;
138  } else {
139  p_thread->p_prev->p_next = p_thread->p_next;
140  p_thread->p_next->p_prev = p_thread->p_prev;
141  if (p_thread == p_cond->p_head) {
142  p_cond->p_head = p_thread->p_next;
143  } else if (p_thread == p_cond->p_tail) {
144  p_cond->p_tail = p_thread->p_prev;
145  }
146  }
147 
148  ABTI_spinlock_release(&p_cond->lock);
149 
150  p_thread->p_prev = NULL;
151  p_thread->p_next = NULL;
152 }

176 int ABT_cond_timedwait(ABT_cond cond, ABT_mutex mutex,
177  const struct timespec *abstime)
178 {
179  int abt_errno = ABT_SUCCESS;

182  ABTI_CHECK_NULL_COND_PTR(p_cond);

184  ABTI_CHECK_NULL_MUTEX_PTR(p_mutex);
185 
186  double tar_time = convert_timespec_to_sec(abstime);
187 
188  ABTI_thread thread;
189  thread.type = ABTI_THREAD_TYPE_EXT;
190  ABTD_atomic_relaxed_store_int(&thread.state, ABTI_THREAD_STATE_BLOCKED);
191 
192  ABTI_spinlock_acquire(&p_cond->lock);
193 
194  if (p_cond->p_waiter_mutex == NULL) {
195  p_cond->p_waiter_mutex = p_mutex;
196  } else {
197  ABT_bool result = ABTI_mutex_equal(p_cond->p_waiter_mutex, p_mutex);
198  if (result == ABT_FALSE) {
199  ABTI_spinlock_release(&p_cond->lock);
200  abt_errno = ABT_ERR_INV_MUTEX;
201  goto fn_fail;
202  }
203  }
204 
205  if (p_cond->num_waiters == 0) {
206  thread.p_prev = &thread;
207  thread.p_next = &thread;
208  p_cond->p_head = &thread;
209  p_cond->p_tail = &thread;
210  } else {
211  p_cond->p_tail->p_next = &thread;
212  p_cond->p_head->p_prev = &thread;
213  thread.p_prev = p_cond->p_tail;
214  thread.p_next = p_cond->p_head;
215  p_cond->p_tail = &thread;
216  }
217

220  ABTI_spinlock_release(&p_cond->lock);
221

223  ABTI_mutex_unlock(p_local_xstream, p_mutex);

228  if (cur_time >= tar_time) {

232  }

241  ABTI_thread_get_ythread(p_local_xstream->p_thread),
242  ABT_SYNC_EVENT_TYPE_COND, (void *)p_cond);
243  }
244

247 
248 fn_exit:
