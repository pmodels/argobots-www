91 static inline size_t unit_get_hash_index(ABT_unit unit)
92 {
93  size_t val = (uintptr_t)unit;
94  /* Let's ignore the first 3 bits and use the next 29 bits. */
95  size_t base_val = val >> 3;
96 #if ABTI_UNIT_HASH_TABLE_SIZE_EXP <= 14
97  base_val += val >> (ABTI_UNIT_HASH_TABLE_SIZE_EXP + 3);
98 #endif
99 #if ABTI_UNIT_HASH_TABLE_SIZE_EXP <= 9
100  base_val += val >> (ABTI_UNIT_HASH_TABLE_SIZE_EXP * 2 + 3);
101 #endif
102  return base_val & (ABTI_UNIT_HASH_TABLE_SIZE - 1);
103 }

168 static void unit_finalize_hash_table(ABTI_global *p_global)
169 {
170  int i;
171  for (i = 0; i < (int)ABTI_UNIT_HASH_TABLE_SIZE; i++) {
172  ABTI_ASSERT(!ABTD_spinlock_is_locked(
173  &p_global->unit_to_thread_entires[i].lock));
174  unit_to_thread *p_cur = atomic_relaxed_load_unit_to_thread(
175  &p_global->unit_to_thread_entires[i].list);
176  while (p_cur) {
177  ABTI_ASSERT(atomic_relaxed_load_unit(&p_cur->unit) ==
178  ABT_UNIT_NULL);
179  unit_to_thread *p_next = p_cur->p_next;
180  ABTU_free(p_cur);
181  p_cur = p_next;
182  }
183  }

186 ABTU_ret_err static inline int
187 unit_map_thread(ABTI_global *p_global, ABT_unit unit, ABTI_thread *p_thread)
188 {
189  ABTI_ASSERT(!ABTI_unit_is_builtin(unit));
190  size_t hash_index = unit_get_hash_index(unit);
191  ABTI_unit_to_thread_entry *p_entry =
192  &p_global->unit_to_thread_entires[hash_index];
193 
194  ABTD_spinlock_acquire(&p_entry->lock);
195  unit_to_thread *p_cur = atomic_relaxed_load_unit_to_thread(&p_entry->list);
196  while (p_cur) {
197  if (atomic_relaxed_load_unit(&p_cur->unit) == ABT_UNIT_NULL) {
198  /* Empty element has been found. Let's use this. */
199  atomic_relaxed_store_unit(&p_cur->unit, unit);
200  /* p_cur is associated with this unit. */
201  p_cur->p_thread = p_thread;
202  ABTD_spinlock_release(&p_entry->lock);
203  return ABT_SUCCESS;
204  }
205  p_cur = p_cur->p_next;
206  }
207  /* It seems that all the elements are in use. Let's allocate a new one. */
208  unit_to_thread *p_new;
209  p_cur = atomic_relaxed_load_unit_to_thread(&p_entry->list);
210  /* Let's dynamically allocate memory. */
211  int ret = ABTU_malloc(sizeof(unit_to_thread), (void **)&p_new);
212  if (ret != ABT_SUCCESS) {
213  ABTD_spinlock_release(&p_entry->lock);
214  return ret;
215  }
216  /* Initialize the new unit. */
217  atomic_relaxed_store_unit(&p_new->unit, unit);
218  p_new->p_thread = p_thread;
219  p_new->p_next = p_cur;
220  atomic_release_store_unit_to_thread(&p_entry->list, p_new);
221  ABTD_spinlock_release(&p_entry->lock);
222  return ABT_SUCCESS;
223 }

225 static inline void unit_unmap_thread(ABTI_global *p_global, ABT_unit unit)
226 {
227  ABTI_ASSERT(!ABTI_unit_is_builtin(unit));
228  size_t hash_index = unit_get_hash_index(unit);
229  ABTI_unit_to_thread_entry *p_entry =
230  &p_global->unit_to_thread_entires[hash_index];
231 
232  ABTD_spinlock_acquire(&p_entry->lock);
233  unit_to_thread *p_cur = atomic_relaxed_load_unit_to_thread(&p_entry->list);
234  /* Update the corresponding unit to "NULL". */
235  while (1) {
236  if (atomic_relaxed_load_unit(&p_cur->unit) == unit) {
237  atomic_relaxed_store_unit(&p_cur->unit, ABT_UNIT_NULL);
238  break;
239  }
240  p_cur = p_cur->p_next;
241  ABTI_ASSERT(p_cur); /* unmap() must succeed. */
242  }
243  ABTD_spinlock_release(&p_entry->lock);
244 }

246 static inline ABTI_thread *
247 unit_get_thread_from_user_defined_unit(ABTI_global *p_global, ABT_unit unit)
248 {
249  ABTI_ASSERT(!ABTI_unit_is_builtin(unit));
250  /* Find an element. */
251  size_t hash_index = unit_get_hash_index(unit);
252  ABTI_unit_to_thread_entry *p_entry =
253  &p_global->unit_to_thread_entires[hash_index];
254  /* The first element must be accessed in a release-acquire manner. The new
255  * element is release-stored to the head, so acquire-load can always get a
256  * valid linked-list chain. */
257  unit_to_thread *p_cur = atomic_acquire_load_unit_to_thread(&p_entry->list);
258  while (1) {
259  ABTI_ASSERT(p_cur); /* get() must succeed. */
260  if (atomic_relaxed_load_unit(&p_cur->unit) == unit) {
261  return p_cur->p_thread;
262  }
263  p_cur = p_cur->p_next;
264  }
