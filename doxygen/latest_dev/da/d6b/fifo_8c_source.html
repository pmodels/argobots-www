195 static ABT_unit pool_pop_wait(ABT_pool pool, double time_secs)
196 {
197  ABTI_pool *p_pool = ABTI_pool_get_ptr(pool);
198  data_t *p_data = pool_get_data_ptr(p_pool->data);
199  unit_t *p_unit = NULL;
200  ABT_unit h_unit = ABT_UNIT_NULL;
201 
202  double time_start = 0.0;
203 
204  do {
205  ABTI_spinlock_acquire(&p_data->mutex);
206  if (p_data->num_units > 0) {
207  p_unit = p_data->p_head;
208  if (p_data->num_units == 1) {
209  p_data->p_head = NULL;
210  p_data->p_tail = NULL;
211  } else {
212  p_unit->p_prev->p_next = p_unit->p_next;
213  p_unit->p_next->p_prev = p_unit->p_prev;
214  p_data->p_head = p_unit->p_next;
215  }
216  p_data->num_units--;
217 
218  p_unit->p_prev = NULL;
219  p_unit->p_next = NULL;
220  ABTD_atomic_release_store_int(&p_unit->is_in_pool, 0);
221 
222  h_unit = (ABT_unit)p_unit;
223  ABTI_spinlock_release(&p_data->mutex);
224  } else {
225  ABTI_spinlock_release(&p_data->mutex);
226  if (time_start == 0.0) {
227  time_start = ABTI_get_wtime();
228  } else {
229  double elapsed = ABTI_get_wtime() - time_start;
230  if (elapsed > time_secs)
231  break;
232  }
233  /* Sleep. */
234  const int sleep_nsecs = 100;
235  struct timespec ts = { 0, sleep_nsecs };
236  nanosleep(&ts, NULL);
237  }
238  } while (h_unit == ABT_UNIT_NULL);
239 
240  return h_unit;
241 }
