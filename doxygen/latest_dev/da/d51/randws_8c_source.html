83  while (1) {

87  ABT_pool pool = p_pools[0];

92  CNT_INC(run_cnt);
93  } else if (num_pools > 1) {

95  target =

103  ABTI_xstream_run_unit(&p_local_xstream, unit, p_pool);
104  CNT_INC(run_cnt);
105  }
106  }
