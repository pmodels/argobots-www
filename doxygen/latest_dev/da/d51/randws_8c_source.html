81 static void sched_run(ABT_sched sched)
82 {
83  ABTI_xstream *p_local_xstream =
84  ABTI_local_get_xstream(ABTI_local_get_local());
85  uint32_t work_count = 0;
86  sched_data *p_data;
87  int num_pools;
88  ABT_pool *pools;
89  ABT_unit unit;
90  int target;
91  unsigned seed = time(NULL);
92  CNT_DECL(run_cnt);
93 
94  ABTI_sched *p_sched = ABTI_sched_get_ptr(sched);
95  ABTI_ASSERT(p_sched);
96 
97  p_data = (sched_data *)p_sched->data;
98  num_pools = p_sched->num_pools;
99  pools = p_data->pools;
100 
101  while (1) {
102  CNT_INIT(run_cnt, 0);
103 
104  /* Execute one work unit from the scheduler's pool */
105  ABT_pool pool = pools[0];
106  ABTI_pool *p_pool = ABTI_pool_get_ptr(pool);
107  unit = ABTI_pool_pop(p_pool);
108  if (unit != ABT_UNIT_NULL) {
109  ABTI_xstream_run_unit(&p_local_xstream, unit, p_pool);
110  CNT_INC(run_cnt);
111  } else if (num_pools > 1) {
112  /* Steal a work unit from other pools */
113  target =
114  (num_pools == 2) ? 1 : (rand_r(&seed) % (num_pools - 1) + 1);
115  pool = pools[target];
116  p_pool = ABTI_pool_get_ptr(pool);
117  unit = ABTI_pool_pop(p_pool);
118  LOG_DEBUG_POOL_POP(p_pool, unit);
119  if (unit != ABT_UNIT_NULL) {
120  ABTI_unit_set_associated_pool(unit, p_pool);
121  ABTI_xstream_run_unit(&p_local_xstream, unit, p_pool);
122  CNT_INC(run_cnt);
123  }
124  }
125 
126  if (++work_count >= p_data->event_freq) {
127  ABTI_xstream_check_events(p_local_xstream, p_sched);
128  ABTI_local *p_local = ABTI_xstream_get_local(p_local_xstream);
129  if (ABTI_sched_has_to_stop(&p_local, p_sched) == ABT_TRUE)
130  break;
131  p_local_xstream = ABTI_local_get_xstream(p_local);
132  work_count = 0;
133  SCHED_SLEEP(run_cnt, p_data->sleep_time);
134  }
135  }
136 }
