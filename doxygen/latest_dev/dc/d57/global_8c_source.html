147 int ABT_finalize(void)
148 {
149  int abt_errno = ABT_SUCCESS;
150  ABTI_local *p_local = ABTI_local_get_local();
151 
152  /* First, take a global lock protecting the initialization/finalization
153  * process. Don't go to fn_exit before taking a lock */
154  ABTI_spinlock_acquire(&g_ABTI_init_lock);
155 
156  /* If Argobots is not initialized, just return */
157  if (g_ABTI_num_inits == 0) {
158  abt_errno = ABT_ERR_UNINITIALIZED;
159  goto fn_exit;
160  }
161  /* If Argobots is still referenced by others, just return */
162  if (--g_ABTI_num_inits != 0)
163  goto fn_exit;
164 
165  ABTI_xstream *p_local_xstream = ABTI_local_get_xstream_or_null(p_local);
166  /* If called by an external thread, return an error. */
167  ABTI_CHECK_TRUE(!ABTI_IS_EXT_THREAD_ENABLED || p_local_xstream,
168  ABT_ERR_INV_XSTREAM);
169 
170  ABTI_CHECK_TRUE_MSG(p_local_xstream->type == ABTI_XSTREAM_TYPE_PRIMARY,
171  ABT_ERR_INV_XSTREAM,
172  "ABT_finalize must be called by the primary ES.");
173 
174  ABTI_thread *p_self = p_local_xstream->p_thread;
175  ABTI_CHECK_TRUE_MSG(p_self->type & ABTI_THREAD_TYPE_MAIN,
176  ABT_ERR_INV_THREAD,
177  "ABT_finalize must be called by the primary ULT.");
178  ABTI_ythread *p_ythread;
179  ABTI_CHECK_YIELDABLE(p_self, &p_ythread, ABT_ERR_INV_THREAD);
180 
181 #ifndef ABT_CONFIG_DISABLE_TOOL_INTERFACE
182  /* Turns off the tool interface */
183  ABTI_tool_event_thread_update_callback(NULL, ABT_TOOL_EVENT_THREAD_NONE,
184  NULL);
185  ABTI_tool_event_task_update_callback(NULL, ABT_TOOL_EVENT_TASK_NONE, NULL);
186 #endif
187 
188  /* Set the join request */
189  ABTI_xstream_set_request(p_local_xstream, ABTI_XSTREAM_REQ_JOIN);
190 
191  /* We wait for the remaining jobs */
192  if (ABTD_atomic_acquire_load_int(&p_local_xstream->state) !=
193  ABT_XSTREAM_STATE_TERMINATED) {
194  /* Set the orphan request for the primary ULT */
195  ABTI_thread_set_request(p_self, ABTI_THREAD_REQ_ORPHAN);
196 
197  LOG_DEBUG("[U%" PRIu64 ":E%d] yield to scheduler\n",
198  ABTI_thread_get_id(p_self),
199  p_ythread->thread.p_last_xstream->rank);
200 
201  /* Switch to the parent */
202  ABTI_ythread_context_switch_to_parent(&p_local_xstream, p_ythread,
203  ABT_SYNC_EVENT_TYPE_XSTREAM_JOIN,
204  (void *)p_local_xstream);
205 
206  /* Back to the original thread */
207  LOG_DEBUG("[U%" PRIu64 ":E%d] resume after yield\n",
208  ABTI_thread_get_id(p_self),
209  p_ythread->thread.p_last_xstream->rank);
210  }
211 
212  /* Remove the primary ULT */
213  ABTI_ASSERT(p_local_xstream->p_thread == p_self);
214  p_local_xstream->p_thread = NULL;
215  ABTI_ythread_free_main(ABTI_xstream_get_local(p_local_xstream), p_ythread);
216 
217  /* Free the primary ES */
218  abt_errno = ABTI_xstream_free(ABTI_xstream_get_local(p_local_xstream),
219  p_local_xstream, ABT_TRUE);
220  ABTI_CHECK_ERROR(abt_errno);
221 
222  /* Finalize the ES local data */
223  ABTI_local_set_xstream(NULL);
224

226  ABTU_free(gp_ABTI_global->p_xstreams);
227

281 void ABTI_global_update_max_xstreams(int new_size)
282 {
283  int i, cur_size, abt_errno;
284 
285  if (new_size != 0 && new_size < gp_ABTI_global->max_xstreams)
286  return;
287 
288  ABTI_spinlock_acquire(&gp_ABTI_global->xstreams_lock);
289 
290  static int max_xstreams_warning_once = 0;
291  if (max_xstreams_warning_once == 0) {
292  /* Because some Argobots functionalities depend on the runtime value
293  * ABT_MAX_NUM_XSTREAMS (or gp_ABTI_global->max_xstreams), changing this
294  * value at run-time can cause an error. For example, using ABT_mutex
295  * created before updating max_xstreams causes an error since
296  * ABTI_thread_htable's array size depends on ABT_MAX_NUM_XSTREAMS.
297  * To fix this issue, please set a larger number to ABT_MAX_NUM_XSTREAMS
298  * in advance. */
299  char *warning_message;
300  abt_errno = ABTU_malloc(sizeof(char) * 1024, (void **)&warning_message);
301  if (!ABTI_IS_ERROR_CHECK_ENABLED || abt_errno == ABT_SUCCESS) {
302  snprintf(warning_message, 1024,
303  "Warning: the number of execution streams exceeds "
304  "ABT_MAX_NUM_XSTREAMS (=%d), which may cause an "
305  "unexpected "
306  "error.",
307  gp_ABTI_global->max_xstreams);
308  HANDLE_WARNING(warning_message);
309  ABTU_free(warning_message);
310  max_xstreams_warning_once = 1;
311  }
312  }
313 
314  cur_size = gp_ABTI_global->max_xstreams;
315  new_size = (new_size > 0) ? new_size : cur_size * 2;
316  gp_ABTI_global->max_xstreams = new_size;
317  abt_errno = ABTU_realloc(cur_size * sizeof(ABTI_xstream *),
318  new_size * sizeof(ABTI_xstream *),
319  (void **)&gp_ABTI_global->p_xstreams);
320  ABTI_ASSERT(abt_errno == ABT_SUCCESS);
321 
322  for (i = gp_ABTI_global->num_xstreams; i < new_size; i++) {
323  gp_ABTI_global->p_xstreams[i] = NULL;
324  }
325 
326  ABTI_spinlock_release(&gp_ABTI_global->xstreams_lock);
327 }
