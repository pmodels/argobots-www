180 ABTU_ret_err static int finailze_library(void)
181 {
182  ABTI_local *p_local = ABTI_local_get_local();
183 
184  /* If Argobots is not initialized, just return */
185  ABTI_CHECK_TRUE(g_ABTI_num_inits > 0, ABT_ERR_UNINITIALIZED);
186  /* If Argobots is still referenced by others, just return */
187  if (--g_ABTI_num_inits != 0) {
188  return ABT_SUCCESS;
189  }
190 
191  ABTI_xstream *p_local_xstream = ABTI_local_get_xstream_or_null(p_local);
192  /* If called by an external thread, return an error. */
193  ABTI_CHECK_TRUE(!ABTI_IS_EXT_THREAD_ENABLED || p_local_xstream,
194  ABT_ERR_INV_XSTREAM);
195 
196  ABTI_CHECK_TRUE_MSG(p_local_xstream->type == ABTI_XSTREAM_TYPE_PRIMARY,
197  ABT_ERR_INV_XSTREAM,
198  "ABT_finalize must be called by the primary ES.");
199 
200  ABTI_thread *p_self = p_local_xstream->p_thread;
201  ABTI_CHECK_TRUE_MSG(p_self->type & ABTI_THREAD_TYPE_PRIMARY,
202  ABT_ERR_INV_THREAD,
203  "ABT_finalize must be called by the primary ULT.");
204  ABTI_ythread *p_ythread;
205  ABTI_CHECK_YIELDABLE(p_self, &p_ythread, ABT_ERR_INV_THREAD);
206 
207 #ifndef ABT_CONFIG_DISABLE_TOOL_INTERFACE
208  /* Turns off the tool interface */
209  ABTI_tool_event_thread_update_callback(NULL, ABT_TOOL_EVENT_THREAD_NONE,
210  NULL);
211 #endif
212 
213  /* Set the orphan request for the primary ULT */
214  ABTI_thread_set_request(p_self, ABTI_THREAD_REQ_ORPHAN);
215  /* Finish the main scheduler of this local xstream. */
216  ABTI_sched_finish(p_local_xstream->p_main_sched);
217  /* p_self cannot join the main scheduler since p_self needs to be orphaned.
218  * Let's wait till the main scheduler finishes. This thread will be
219  * scheduled when the main root thread finishes. */
220  ABTI_ythread_yield(&p_local_xstream, p_ythread, ABT_SYNC_EVENT_TYPE_OTHER,
221  NULL);
222  ABTI_ASSERT(p_local_xstream == ABTI_local_get_xstream(p_local));
223  ABTI_ASSERT(p_local_xstream->p_thread == p_self);
224 
225  /* Remove the primary ULT */
226  p_local_xstream->p_thread = NULL;
227  ABTI_ythread_free_primary(ABTI_xstream_get_local(p_local_xstream),
228  p_ythread);
229

231  ABTI_xstream_free(ABTI_xstream_get_local(p_local_xstream), p_local_xstream,
232  ABT_TRUE);
233

235  ABTI_local_set_xstream(NULL);
236

239

242

245  ABTD_affinity_finalize();
246  }
247

249  ABTU_free(gp_ABTI_global);
250  gp_ABTI_global = NULL;
251  ABTD_atomic_release_store_uint32(&g_ABTI_initialized, 0);
