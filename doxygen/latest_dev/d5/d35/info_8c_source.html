690 int ABT_info_trigger_print_all_thread_stacks(FILE *fp, double timeout,
691  void (*cb_func)(ABT_bool, void *),
692  void *arg)
693 {
694  /* This function is signal-safe, so it may not call other functions unless
695  * you really know what the called functions do. */
696  if (ABTD_atomic_acquire_load_uint32(&print_stack_flag) ==
697  PRINT_STACK_FLAG_UNSET) {
698  if (ABTD_atomic_bool_cas_strong_uint32(&print_stack_flag,
699  PRINT_STACK_FLAG_UNSET,
700  PRINT_STACK_FLAG_INITIALIZE)) {
701  /* Save fp and timeout. */
702  print_stack_fp = fp;
703  print_stack_timeout = timeout;
704  print_cb_func = cb_func;
705  print_arg = arg;
706  /* Here print_stack_barrier must be 0. */
707  ABTI_ASSERT(ABTD_atomic_acquire_load_uint32(&print_stack_barrier) ==
708  0);
709  ABTD_atomic_release_store_uint32(&print_stack_flag,
710  PRINT_STACK_FLAG_WAIT);
711  }
712  }
713  return ABT_SUCCESS;
714 }

716 void ABTI_info_check_print_all_thread_stacks(void)
717 {
718  if (ABTD_atomic_acquire_load_uint32(&print_stack_flag) !=
719  PRINT_STACK_FLAG_WAIT)
720  return;
721 
722  /* Wait for the other execution streams using a barrier mechanism. */
723  uint32_t self_value = ABTD_atomic_fetch_add_uint32(&print_stack_barrier, 1);
724  if (self_value == 0) {
725  /* This ES becomes the main ES. */
726  double start_time = ABTI_get_wtime();
727  ABT_bool force_print = ABT_FALSE;
728 
729  /* xstreams_lock is acquired to avoid dynamic ES creation while
730  * printing data. */
731  ABTI_spinlock_acquire(&gp_ABTI_global->xstreams_lock);
732  while (1) {
733  if (ABTD_atomic_acquire_load_uint32(&print_stack_barrier) >=
734  gp_ABTI_global->num_xstreams) {
735  break;
736  }
737  if (print_stack_timeout >= 0.0 &&
738  (ABTI_get_wtime() - start_time) >= print_stack_timeout) {
739  force_print = ABT_TRUE;
740  break;
741  }
742  ABTI_spinlock_release(&gp_ABTI_global->xstreams_lock);
743  ABTD_atomic_pause();
744  ABTI_spinlock_acquire(&gp_ABTI_global->xstreams_lock);
745  }
746  /* All the available ESs are (supposed to be) stopped. We *assume* that
747  * no ES is calling and will call Argobots functions except this
748  * function while printing stack information. */
749  int i, j;
750  struct ABTI_info_pool_set_t pool_set;
751  ABTI_info_initialize_pool_set(&pool_set);
752  FILE *fp = print_stack_fp;
753  if (force_print) {
754  fprintf(fp,
755  "ABT_info_trigger_print_all_thread_stacks: "
756  "timeout (only %d ESs stop)\n",
757  (int)ABTD_atomic_acquire_load_uint32(&print_stack_barrier));
758  }
759  for (i = 0; i < gp_ABTI_global->num_xstreams; i++) {
760  ABTI_xstream *p_xstream = gp_ABTI_global->p_xstreams[i];
761  ABTI_sched *p_main_sched = p_xstream->p_main_sched;
762  fprintf(fp, "= xstream[%d] (%p) =\n", i, (void *)p_xstream);
763  fprintf(fp, "main_sched : %p\n", (void *)p_main_sched);
764  if (!p_main_sched)
765  continue;
766  for (j = 0; j < p_main_sched->num_pools; j++) {
767  ABT_pool pool = p_main_sched->pools[j];
768  ABTI_ASSERT(pool != ABT_POOL_NULL);
769  fprintf(fp, " pools[%d] : %p\n", j,
770  (void *)ABTI_pool_get_ptr(pool));
771  ABTI_info_add_pool_set(pool, &pool_set);
772  }
773  }
774  for (i = 0; i < pool_set.num; i++) {
775  ABT_pool pool = pool_set.pools[i];
776  ABTI_pool *p_pool = ABTI_pool_get_ptr(pool);
777  int abt_errno = ABTI_info_print_thread_stacks_in_pool(fp, p_pool);
778  if (abt_errno != ABT_SUCCESS)
779  fprintf(fp, " Failed to print (errno = %d).\n", abt_errno);
780  }
781  /* Release the lock that protects ES data. */
782  ABTI_spinlock_release(&gp_ABTI_global->xstreams_lock);
783 
784  if (print_cb_func)
785  print_cb_func(force_print, print_arg);
786  ABTI_info_finalize_pool_set(&pool_set);
787  /* Update print_stack_flag to 3. */
788  ABTD_atomic_release_store_uint32(&print_stack_flag,
789  PRINT_STACK_FLAG_FINALIZE);
790  } else {
791  /* Wait for the main ES's work. */
792  while (ABTD_atomic_acquire_load_uint32(&print_stack_flag) !=
793  PRINT_STACK_FLAG_FINALIZE)
794  ABTD_atomic_pause();
795  }
796  ABTI_ASSERT(ABTD_atomic_acquire_load_uint32(&print_stack_flag) ==
797  PRINT_STACK_FLAG_FINALIZE);
798 
799  /* Decrement the barrier value. */
800  uint32_t dec_value = ABTD_atomic_fetch_sub_uint32(&print_stack_barrier, 1);
801  if (dec_value == 0) {
802  /* The last execution stream resets the flag. */
803  ABTD_atomic_release_store_uint32(&print_stack_flag,
804  PRINT_STACK_FLAG_UNSET);
805  }
806 }
