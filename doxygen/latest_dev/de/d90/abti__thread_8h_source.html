47 #if ABT_CONFIG_THREAD_TYPE == ABT_THREAD_TYPE_DYNAMIC_PROMOTION
48 static inline ABT_bool ABTI_thread_is_dynamic_promoted(ABTI_thread *p_thread)
49 {
50  /*
51  * Create a context and switch to it. The flow of the dynamic promotion
52  * thread is as follows:
53  *
54  * - When a ULT does not yield:
55  * ABTI_xstream_schedule_thread : call init_and_call_fcontext
56  * init_and_call_fcontext : jump to the stack top
57  * : save the scheduler's context
58  * : call the thread function
59  * thread_f : start thread_f
60  * : [ULT body]
61  * : end thread_f
62  * init_and_call_fcontext : calculate the return address, which is
63  * [1] => the original return address.
64  * : `return`
65  * ABTI_xstream_schedule_thread : resume the scheduler
66  *
67  * The ULT can return to the original scheduler by `return` if the scheduler
68  * has never been resumed during the execution of the ULT, because the
69  * context of the parent scheduler can be restored in a normal return
70  * procedure. In this case, the context is saved only once.
71  *
72  * - When a ULT yields:
73  * ABTI_xstream_schedule_thread : call init_and_call_fcontext
74  * init_and_call_fcontext : jump to the stack top
75  * : save the scheduler's context
76  * : call the thread function
77  * thread_f : start thread_f
78  * : [yield in ULT body]
79  * ABTD_thread_context_dynamic_promote
80  * : rewrite the return address to
81  * ABTD_thread_terminate_thread_no_arg
82  * jump_fcontext : save the ULT's context
83  * : restore the scheduler's context
84  * : jump to the scheduler
85  * ABTI_xstream_schedule_thread : resume the scheduler
86  *
87  * ... After a while, a scheduler resumes this ULT ...
88  *
89  * jump_fcontext : save the scheduler's context
90  * : restore the ULT's context
91  * : jump to the ULT
92  * thread_f : [ULT body (remaining)]
93  * : end thread_f
94  * init_and_call_fcontext : calculate the return address, which is
95  * [2] => ABTD_thread_terminate_thread_no_arg
96  * : return
97  * ABTD_thread_terminate_thread_no_arg
98  * : call take_fcontext
99  * take_fcontext : restore the scheduler's context
100  * : jump to the scheduler
101  * ABTI_xstream_schedule_thread : resume the scheduler
102  *
103  * When a ULT yields, ABTD_thread_terminate_thread_no_arg is set to
104  * [ptr - 0x08] so that it can "jump" to the normal termination
105  * function by "return" in init_and_call_fcontext. This termination
106  * function calls take_fcontext, so the scheduler is resumed by user-level
107  * context switch.
108  *
109  * For example, the stack will be as follows at [1] and [2] in the x86-64
110  * case. Note that ptr points to the stack top (= p_stack + stacksize).
111  *
112  * In the case of [1] (no suspension):
113  * [0x12345600] : (the original instruction pointer)
114  * ...
115  * [ptr - 0x08] : the original stack pointer (i.e., 0x12345600)
116  * [ptr - 0x10] : unused (for 16-byte alignment)
117  * [ptr - xxxx] : used by thread_f
118  *
119  * In the case of [2] (after suspension):
120  * [ptr - 0x08] : pointing to (p_stack - 0x10)
121  * [ptr - 0x10] : the address of ABTD_thread_terminate_thread_no_arg
122  * [ptr - xxxx] : used by thread_f
123  *
124  * This technique was introduced as a "return-on-completion" thread in the
125  * following paper:
126  * Lessons Learned from Analyzing Dynamic Promotion for User-Level
127  * Threading, S. Iwasaki, A. Amer, K. Taura, and P. Balaji (SC '18)
128  */
129  return ABTD_thread_context_is_dynamic_promoted(&p_thread->ctx);
130 }

143 static inline ABTI_thread *ABTI_thread_context_switch_to_sibling_internal(
144  ABTI_xstream **pp_local_xstream, ABTI_thread *p_old, ABTI_thread *p_new,
145  ABT_bool is_finish)
146 {
147 #if ABT_CONFIG_THREAD_TYPE == ABT_THREAD_TYPE_DYNAMIC_PROMOTION
148  /* Dynamic promotion is unnecessary if p_old will be discarded. */
149  if (!ABTI_thread_is_dynamic_promoted(p_old)) {
150  ABTI_thread_dynamic_promote_thread(p_old);
151  }
152  if (!ABTI_thread_is_dynamic_promoted(p_new)) {
153  /* p_new does not have a context, so we first need to make it. */
154  ABTD_thread_context_arm_thread(p_new->stacksize, p_new->p_stack,
155  &p_new->ctx);
156  }
157 #endif
158  p_new->unit_def.p_parent = p_old->unit_def.p_parent;
159  if (is_finish) {
160  ABTI_tool_event_thread_finish(*pp_local_xstream, p_old,
161  p_old->unit_def.p_parent);
162  ABTD_thread_finish_context(&p_old->ctx, &p_new->ctx);
163  return NULL; /* Unreachable. */
164  } else {
165  ABTD_thread_context_switch(&p_old->ctx, &p_new->ctx);
166  ABTI_xstream *p_local_xstream = ABTI_local_get_xstream_uninlined();
167  *pp_local_xstream = p_local_xstream;
168  ABTI_unit *p_prev = p_local_xstream->p_unit;
169  p_local_xstream->p_unit = &p_old->unit_def;
170  ABTI_ASSERT(ABTI_unit_type_is_thread(p_prev->type));
171  return ABTI_unit_get_thread(p_prev);
172  }

210 static inline ABTI_thread *ABTI_thread_context_switch_to_child_internal(
211  ABTI_xstream **pp_local_xstream, ABTI_thread *p_old, ABTI_thread *p_new)
212 {
213  ABTI_xstream *p_local_xstream;
214  p_new->unit_def.p_parent = &p_old->unit_def;
215 #if ABT_CONFIG_THREAD_TYPE == ABT_THREAD_TYPE_DYNAMIC_PROMOTION
216  if (!ABTI_thread_is_dynamic_promoted(p_old)) {
217  ABTI_thread_dynamic_promote_thread(p_old);
218  }
219  if (!ABTI_thread_is_dynamic_promoted(p_new)) {
220  void *p_stacktop = ((char *)p_new->p_stack) + p_new->stacksize;
221  LOG_DEBUG("[U%" PRIu64 "] run ULT (dynamic promotion)\n",
222  ABTI_thread_get_id(p_new));
223  p_local_xstream = *pp_local_xstream;
224  p_local_xstream->p_unit = &p_new->unit_def;
225  /* Invoke an event of thread run. */
226  ABTI_tool_event_thread_run(p_local_xstream, p_new, &p_old->unit_def,
227  &p_old->unit_def);
228  ABTD_thread_context_make_and_call(&p_old->ctx, p_new->unit_def.f_unit,
229  p_new->unit_def.p_arg, p_stacktop);
230  /* The scheduler continues from here. If the previous thread has not
231  * run dynamic promotion, ABTI_thread_context_make_and_call took the
232  * fast path. In this case, the request handling has not been done,
233  * so it must be done here. */
234  p_local_xstream = ABTI_local_get_xstream_uninlined();
235  *pp_local_xstream = p_local_xstream;
236  ABTI_unit *p_prev_unit = p_local_xstream->p_unit;
237  ABTI_ASSERT(ABTI_unit_type_is_thread(p_prev_unit->type));
238  ABTI_thread *p_prev = ABTI_unit_get_thread(p_prev_unit);
239  p_local_xstream->p_unit = &p_old->unit_def;
240  if (!ABTI_thread_is_dynamic_promoted(p_prev)) {
241  ABTI_ASSERT(p_prev == p_new);
242  /* Invoke a thread-finish event of the previous thread. */
243  ABTI_tool_event_thread_finish(p_local_xstream, p_prev,
244  &p_old->unit_def);
245  /* See ABTDI_thread_terminate for details.
246  * TODO: avoid making a copy of the code. */
247  ABTD_thread_context *p_ctx = &p_prev->ctx;
248  ABTD_thread_context *p_link =
249  ABTD_atomic_acquire_load_thread_context_ptr(&p_ctx->p_link);
250  if (p_link) {
251  /* If p_link is set, it means that other ULT has called the
252  * join. */
253  ABTI_thread *p_joiner = ABTI_thread_context_get_thread(p_link);
254  /* The scheduler may not use a bypass mechanism, so just makes
255  * p_joiner ready. */
256  ABTI_thread_set_ready(p_local_xstream, p_joiner);
257 
258  /* We don't need to use the atomic OR operation here because
259  * the ULT will be terminated regardless of other requests. */
260  ABTD_atomic_release_store_uint32(&p_prev->unit_def.request,
261  ABTI_UNIT_REQ_TERMINATE);
262  } else {
263  uint32_t req =
264  ABTD_atomic_fetch_or_uint32(&p_prev->unit_def.request,
265  ABTI_UNIT_REQ_JOIN |
266  ABTI_UNIT_REQ_TERMINATE);
267  if (req & ABTI_UNIT_REQ_JOIN) {
268  /* This case means there has been a join request and the
269  * joiner has blocked. We have to wake up the joiner ULT.
270  */
271  do {
272  p_link = ABTD_atomic_acquire_load_thread_context_ptr(
273  &p_ctx->p_link);
274  } while (!p_link);
275  ABTI_thread_set_ready(p_local_xstream,
276  ABTI_thread_context_get_thread(
277  p_link));
278  }
279  }
280  }
281  return p_prev;
282  }
283 #endif
284  {
285  ABTD_thread_context_switch(&p_old->ctx, &p_new->ctx);
286  p_local_xstream = ABTI_local_get_xstream_uninlined();
287  *pp_local_xstream = p_local_xstream;
288  ABTI_unit *p_prev = p_local_xstream->p_unit;
289  p_local_xstream->p_unit = &p_old->unit_def;
290  ABTI_ASSERT(ABTI_unit_type_is_thread(p_prev->type));
291  /* p_old keeps running as a parent, so no thread-run event incurs. */
292  return ABTI_unit_get_thread(p_prev);
293  }
294 }
