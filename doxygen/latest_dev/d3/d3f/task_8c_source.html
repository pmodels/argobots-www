552 ABTU_ret_err static int task_create(ABTI_local *p_local, ABTI_pool *p_pool,
553  void (*task_func)(void *), void *arg,
554  ABTI_sched *p_sched, int refcount,
555  ABTI_thread **pp_newtask)
556 {
557  ABTI_thread *p_newtask;
558  ABT_task h_newtask;
559 
560  /* Allocate a task object */
561  int abt_errno = ABTI_mem_alloc_nythread(p_local, &p_newtask);
562  ABTI_CHECK_ERROR(abt_errno);
563 
564  p_newtask->p_last_xstream = NULL;
565  p_newtask->p_parent = NULL;
566  ABTD_atomic_relaxed_store_int(&p_newtask->state, ABT_THREAD_STATE_READY);
567  ABTD_atomic_relaxed_store_uint32(&p_newtask->request, 0);
568  p_newtask->f_thread = task_func;
569  p_newtask->p_arg = arg;
570  p_newtask->p_pool = p_pool;
571  ABTD_atomic_relaxed_store_ptr(&p_newtask->p_keytable, NULL);
572  p_newtask->id = ABTI_TASK_INIT_ID;
573 
574  /* Create a wrapper work unit */
575  h_newtask = ABTI_thread_get_handle(p_newtask);
576  ABTI_thread_type thread_type =
577  refcount ? (ABTI_THREAD_TYPE_THREAD | ABTI_THREAD_TYPE_NAMED)
578  : ABTI_THREAD_TYPE_THREAD;
579 #ifndef ABT_CONFIG_DISABLE_MIGRATION
580  thread_type |= ABTI_THREAD_TYPE_MIGRATABLE;
581 #endif
582  p_newtask->type |= thread_type;
583  p_newtask->unit = p_pool->u_create_from_thread(h_newtask);
584 
585  ABTI_tool_event_thread_create(p_local, p_newtask,
586  ABTI_local_get_xstream_or_null(p_local)
587  ? ABTI_local_get_xstream(p_local)
588  ->p_thread
589  : NULL,
590  p_pool);
591  LOG_DEBUG("[T%" PRIu64 "] created\n", ABTI_thread_get_id(p_newtask));
592 
593  /* Add this task to the scheduler's pool */
594  ABTI_pool_push(p_pool, p_newtask->unit);
595 
596  /* Return value */
597  *pp_newtask = p_newtask;
598 
599  return ABT_SUCCESS;
