2067 static inline int thread_join(ABTI_local **pp_local, ABTI_thread *p_thread)
2068 {

2070  ABT_THREAD_STATE_TERMINATED) {
2071  goto fn_exit;
2072  }
2073 
2074  /* The main ULT cannot be joined. */
2075  ABTI_CHECK_TRUE_RET(!(p_thread->type & (ABTI_THREAD_TYPE_MAIN |
2076  ABTI_THREAD_TYPE_MAIN_SCHED)),
2077  ABT_ERR_INV_THREAD);
2078 
2079  ABTI_xstream *p_local_xstream = ABTI_local_get_xstream_or_null(*pp_local);
2080  if (ABTI_IS_EXT_THREAD_ENABLED && !p_local_xstream)
2081  goto busywait_based;
2082 
2083  ABTI_thread *p_self_thread = p_local_xstream->p_thread;
2084  /* The target ULT should be different. */
2085  ABTI_CHECK_TRUE_RET(p_thread != p_self_thread, ABT_ERR_INV_THREAD);
2086 
2087  ABTI_ythread *p_self = ABTI_thread_get_ythread_or_null(p_self_thread);
2088  if (!p_self)
2089  goto busywait_based;
2090 
2091  ABTI_ythread *p_ythread = ABTI_thread_get_ythread_or_null(p_thread);
2092  if (!p_ythread)
2093  goto yield_based_task;
2094 
2095  ABT_pool_access access = p_self->thread.p_pool->access;
2096 
2097  if ((p_self->thread.p_pool == p_ythread->thread.p_pool) &&
2098  (access == ABT_POOL_ACCESS_PRIV || access == ABT_POOL_ACCESS_MPSC ||
2099  access == ABT_POOL_ACCESS_SPSC) &&
2100  (ABTD_atomic_acquire_load_int(&p_ythread->thread.state) ==
2101  ABT_THREAD_STATE_READY)) {
2102 
2103  ABTI_xstream *p_xstream = p_self->thread.p_last_xstream;
2104 
2105  /* If other ES is calling ABTI_ythread_set_ready(), p_ythread may not
2106  * have been added to the pool yet because ABTI_ythread_set_ready()
2107  * changes the state first followed by pushing p_ythread to the pool.
2108  * Therefore, we have to check whether p_ythread is in the pool, and if
2109  * not, we need to wait until it is added. */
2110  while (p_ythread->thread.p_pool->u_is_in_pool(p_ythread->thread.unit) !=
2111  ABT_TRUE) {
2112  }
2113 
2114  /* This is corresponding to suspension. */
2115  ABTI_tool_event_ythread_suspend(p_local_xstream, p_self,
2116  p_self->thread.p_parent,
2117  ABT_SYNC_EVENT_TYPE_THREAD_JOIN,
2118  (void *)p_ythread);
2119 
2120  /* Increase the number of blocked units. Be sure to execute
2121  * ABTI_pool_inc_num_blocked before ABTI_POOL_REMOVE in order not to
2122  * underestimate the number of units in a pool. */
2123  ABTI_pool_inc_num_blocked(p_self->thread.p_pool);
2124  /* Remove the target ULT from the pool */
2125  int abt_errno =
2126  ABTI_pool_remove(ABTI_xstream_get_local(p_local_xstream),
2127  p_ythread->thread.p_pool, p_ythread->thread.unit);
2128  ABTI_CHECK_ERROR_RET(abt_errno);
2129 
2130  /* Set the link in the context for the target ULT. Since p_link will be
2131  * referenced by p_self, this update does not require release store. */
2132  ABTD_atomic_relaxed_store_ythread_context_ptr(&p_ythread->ctx.p_link,
2133  &p_self->ctx);
2134  /* Set the last ES */
2135  p_ythread->thread.p_last_xstream = p_xstream;
2136  ABTD_atomic_release_store_int(&p_ythread->thread.state,
2137  ABT_THREAD_STATE_RUNNING);
2138 
2139  /* Make the current ULT BLOCKED */
2140  ABTD_atomic_release_store_int(&p_self->thread.state,
2141  ABT_THREAD_STATE_BLOCKED);
2142 
2143  LOG_DEBUG("[U%" PRIu64 ":E%d] blocked to join U%" PRIu64 "\n",
2144  ABTI_thread_get_id(&p_self->thread),
2145  p_self->thread.p_last_xstream->rank,
2146  ABTI_thread_get_id(&p_ythread->thread));
2147  LOG_DEBUG("[U%" PRIu64 ":E%d] start running\n",
2148  ABTI_thread_get_id(&p_ythread->thread),
2149  p_ythread->thread.p_last_xstream->rank);
2150 
2151  /* Switch the context */
2152  ABTI_ythread *p_prev =
2153  ABTI_ythread_context_switch_to_sibling(&p_local_xstream, p_self,
2154  p_ythread);
2155  *pp_local = ABTI_xstream_get_local(p_local_xstream);
2156  ABTI_tool_event_thread_run(p_local_xstream, &p_self->thread,
2157  &p_prev->thread, p_self->thread.p_parent);
2158 
2159  } else if ((p_self->thread.p_pool != p_ythread->thread.p_pool) &&
2160  (access == ABT_POOL_ACCESS_PRIV ||
2161  access == ABT_POOL_ACCESS_SPSC)) {
2162  /* FIXME: once we change the suspend/resume mechanism (i.e., asking the
2163  * scheduler to wake up the blocked ULT), we will be able to handle all
2164  * access modes. */
2165  goto yield_based;
2166 
2167  } else {
2168  /* Tell p_ythread that there has been a join request. */
2169  /* If request already has ABTI_THREAD_REQ_JOIN, p_ythread is
2170  * terminating. We can't block p_self in this case. */
2171  uint32_t req = ABTD_atomic_fetch_or_uint32(&p_ythread->thread.request,
2172  ABTI_THREAD_REQ_JOIN);
2173  if (req & ABTI_THREAD_REQ_JOIN)
2174  goto yield_based;
2175

2177  LOG_DEBUG("[U%" PRIu64 ":E%d] blocked to join U%" PRIu64 "\n",
2178  ABTI_thread_get_id(&p_self->thread),
2179  p_self->thread.p_last_xstream->rank,
2180  ABTI_thread_get_id(&p_ythread->thread));
2181 
2182  /* Set the link in the context of the target ULT. This p_link might be
