153 static inline void ABTI_mem_free_ythread(ABTI_local *p_local,
154  ABTI_ythread *p_ythread)
155 {
156  /* Return stack. */
157 #ifdef ABT_CONFIG_USE_MEM_POOL
158  if (p_ythread->stacktype == ABTI_STACK_TYPE_MEMPOOL) {
159  ABTI_VALGRIND_UNREGISTER_STACK(p_ythread->p_stack);
160  ABTI_xstream *p_local_xstream = ABTI_local_get_xstream_or_null(p_local);
161  /* Came from a memory pool. */
162 #ifndef ABT_CONFIG_DISABLE_EXT_THREAD
163  if (p_local_xstream == NULL) {
164  /* Return a stack to the global pool. */
165  ABTI_spinlock_acquire(&gp_ABTI_global->mem_pool_stack_lock);
166  ABTI_mem_pool_free(&gp_ABTI_global->mem_pool_stack_ext, p_ythread);
167  ABTI_spinlock_release(&gp_ABTI_global->mem_pool_stack_lock);
168  return;
169  }
170 #endif
171  ABTI_mem_pool_free(&p_local_xstream->mem_pool_stack, p_ythread);
172  } else
173 #endif
174  if (p_ythread->stacktype == ABTI_STACK_TYPE_MALLOC) {
175  ABTI_VALGRIND_UNREGISTER_STACK(p_ythread->p_stack);
176  /* p_ythread is allocated together with the stack. */
177  ABTU_free(p_ythread->p_stack);
178  } else {
179  if (p_ythread->stacktype == ABTI_STACK_TYPE_USER) {
180  ABTI_VALGRIND_UNREGISTER_STACK(p_ythread->p_stack);
181  }
182  ABTU_free(p_ythread);
183  }
184 }
